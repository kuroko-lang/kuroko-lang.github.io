34  case VAL_BOOLEAN: fprintf(f, "%s", AS_BOOLEAN(printable) ? "True" : "False"); break;
35  case VAL_FLOATING: fprintf(f, "%g", AS_FLOATING(printable)); break;
36  case VAL_NONE: fprintf(f, "None"); break;
37  case VAL_HANDLER: fprintf(f, "{%s->%d}", AS_HANDLER(printable).type == OP_PUSH_TRY ? "try" : "with", (int)AS_HANDLER(printable).target); break;
38  case VAL_KWARGS: {
39  if (AS_INTEGER(printable) == LONG_MAX) {
40  fprintf(f, "{unpack single}");
41  } else if (AS_INTEGER(printable) == LONG_MAX-1) {
42  fprintf(f, "{unpack list}");
43  } else if (AS_INTEGER(printable) == LONG_MAX-2) {
44  fprintf(f, "{unpack dict}");
45  } else if (AS_INTEGER(printable) == LONG_MAX-3) {
46  fprintf(f, "{unpack nil}");
47  } else if (AS_INTEGER(printable) == 0) {
48  fprintf(f, "{unset default}");
49  } else {
50  fprintf(f, "{sentinel=" PRIkrk_int "}",AS_INTEGER(printable));
51  }
52  break;
53  }
54  default: break;
55  }
56  return;
57  }

59  if (type->_tostr) {

76 void krk_printValueSafe(FILE * f, KrkValue printable) {
77  if (!IS_OBJECT(printable)) {
78  krk_printValue(f,printable);
79  } else if (IS_STRING(printable)) {
80  fprintf(f, "'");
81  /*
82  * Print at most STRING_DEBUG_TRUNCATE characters, as bytes, escaping anything not ASCII.
83  * See also str.__repr__ which does something similar with escape sequences, but this
84  * is a dumber, safer, and slightly faster approach.
85  */
86  for (size_t c = 0; c < AS_STRING(printable)->length && c < STRING_DEBUG_TRUNCATE; ++c) {
87  unsigned char byte = (unsigned char)AS_CSTRING(printable)[c];
88  switch (byte) {
89  case '\\': fprintf(f, "\\\\"); break;
90  case '\n': fprintf(f, "\\n"); break;
91  case '\r': fprintf(f, "\\r"); break;
92  case '\'': fprintf(f, "\\'"); break;
93  default: {
94  if (byte < ' ' || byte > '~') {
95  fprintf(f, "\\x%02x", byte);
96  } else {
97  fprintf(f, "%c", byte);
98  }
99  break;
100  }
101  }
102  }
103  if (AS_STRING(printable)->length > STRING_DEBUG_TRUNCATE) {
104  fprintf(f,"...");
105  }
106  fprintf(f,"'");
107  } else {
108  switch (AS_OBJECT(printable)->type) {
109  case OBJ_FUNCTION: fprintf(f, "<function %s>", AS_FUNCTION(printable)->name ? AS_FUNCTION(printable)->name->chars : "?"); break;
110  case OBJ_CLASS: fprintf(f, "<class %s>", AS_CLASS(printable)->name ? AS_CLASS(printable)->name->chars : "?"); break;
111  case OBJ_INSTANCE: fprintf(f, "<instance of %s>", AS_INSTANCE(printable)->_class->name->chars); break;
112  case OBJ_NATIVE: fprintf(f, "<nativefn %s>", ((KrkNative*)AS_OBJECT(printable))->name); break;
113  case OBJ_CLOSURE: fprintf(f, "<function %s>", AS_CLOSURE(printable)->function->name->chars); break;
114  case OBJ_BYTES: fprintf(f, "<bytes of len %ld>", (long)AS_BYTES(printable)->length); break;
115  case OBJ_TUPLE: {
116  fprintf(f, "<tuple (");
117  for (size_t i = 0; i < AS_TUPLE(printable)->values.count; ++i) {
118  krk_printValueSafe(f, AS_TUPLE(printable)->values.values[i]);
119  }
120  fprintf(f, ")>");
121  } break;
122  case OBJ_BOUND_METHOD: fprintf(f, "<method %s>",
123  AS_BOUND_METHOD(printable)->method ? (
124  AS_BOUND_METHOD(printable)->method->type == OBJ_CLOSURE ? ((KrkClosure*)AS_BOUND_METHOD(printable)->method)->function->name->chars :
125  ((KrkNative*)AS_BOUND_METHOD(printable)->method)->name) : "(corrupt bound method)"); break;
126  default: fprintf(f, "<%s>", krk_typeName(printable)); break;
127  }
128  }
129 }

139 int krk_valuesEqual(KrkValue a, KrkValue b) {
140  if (a.type == b.type) {
141  switch (a.type) {
142  case VAL_BOOLEAN: return AS_BOOLEAN(a) == AS_BOOLEAN(b);
143  case VAL_NONE: return 1; /* None always equals None */
144  case VAL_KWARGS: /* Equal if same number of args; may be useful for comparing sentinels (0) to arg lists. */
145  case VAL_INTEGER: return AS_INTEGER(a) == AS_INTEGER(b);
146  case VAL_FLOATING: return AS_FLOATING(a) == AS_FLOATING(b);
147  case VAL_HANDLER: krk_runtimeError(vm.exceptions->valueError,"Invalid value"); return 0;
148  case VAL_OBJECT: {
149  if (AS_OBJECT(a) == AS_OBJECT(b)) return 1;
150  } break;
151  default: break;
152  }
153  }
154 
155  if (IS_KWARGS(a) || IS_KWARGS(b)) return 0;
156 
157  if (!IS_OBJECT(a) && !IS_OBJECT(b)) {
158  switch (a.type) {
159  case VAL_INTEGER: {
160  switch (b.type) {
161  case VAL_BOOLEAN: return AS_INTEGER(a) == AS_BOOLEAN(b);
162  case VAL_FLOATING: return (double)AS_INTEGER(a) == AS_FLOATING(b);
163  default: return 0;
164  }
165  } break;
166  case VAL_FLOATING: {
167  switch (b.type) {
168  case VAL_BOOLEAN: return AS_FLOATING(a) == (double)AS_BOOLEAN(b);
169  case VAL_INTEGER: return AS_FLOATING(a) == (double)AS_INTEGER(b);
170  default: return 0;
171  }
172  } break;
173  case VAL_BOOLEAN: {
174  switch (b.type) {
175  case VAL_INTEGER: return AS_BOOLEAN(a) == AS_INTEGER(b);
176  case VAL_FLOATING: return (double)AS_BOOLEAN(a) == AS_FLOATING(b);
177  default: return 0;
178  }
179  } break;
180  default: return 0;
181  }
182  }
183 
184  if (IS_TUPLE(a) && IS_TUPLE(b)) {
185  KrkTuple * self = AS_TUPLE(a);
186  KrkTuple * them = AS_TUPLE(b);
187  if (self->values.count != them->values.count || self->obj.hash != them->obj.hash) return 0;
188  for (size_t i = 0; i < self->values.count; ++i) {
189  if (!krk_valuesEqual(self->values.values[i], them->values.values[i])) return 0;
190  }
191  return 1;
192  }

195  if (type && type->_eq) {
196  krk_push(a);
197  krk_push(b);
198  KrkValue result = krk_callSimple(OBJECT_VAL(type->_eq),2,0);
199  if (IS_BOOLEAN(result)) return AS_BOOLEAN(result);
200  return 0;
