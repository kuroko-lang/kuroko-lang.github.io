858 static int do_bin_op(KrkLong * res, const KrkLong * a, const KrkLong * b, char op) {
859  size_t awidth = a->width < 0 ? -a->width : a->width;
860  size_t bwidth = b->width < 0 ? -b->width : b->width;
861  size_t owidth = ((awidth > bwidth) ? awidth : bwidth) + 1;
862 
863  int aneg = (a->width < 0);
864  int bneg = (b->width < 0);
865  int rneg = 0;
866 
867  switch (op) {
868  case '|': rneg = aneg | bneg; break;
869  case '^': rneg = aneg ^ bneg; break;
870  case '&': rneg = aneg & bneg; break;
871  }
872 
873  krk_long_resize(res, owidth);
874 
875  int acarry = aneg ? 1 : 0;
876  int bcarry = bneg ? 1 : 0;
877  int rcarry = rneg ? 1 : 0;
878 
879  for (size_t i = 0; i < owidth; ++i) {
880  uint32_t a_digit = (i < awidth ? a->digits[i] : 0);
881  a_digit = aneg ? ((a_digit ^ DIGIT_MAX) + acarry) : a_digit;
882  acarry = a_digit >> DIGIT_SHIFT;
883 
884  uint32_t b_digit = (i < bwidth ? b->digits[i] : 0);
885  b_digit = bneg ? ((b_digit ^ DIGIT_MAX) + bcarry) : b_digit;
886  bcarry = b_digit >> DIGIT_SHIFT;
887 
888  uint32_t r;
889  switch (op) {
890  case '|': r = a_digit | b_digit; break;
891  case '^': r = a_digit ^ b_digit; break;
892  case '&': r = a_digit & b_digit; break;
893  default: __builtin_unreachable();
894  }
895 
896  r = rneg ? (((r & DIGIT_MAX) ^ DIGIT_MAX) + rcarry) : r;
897  res->digits[i] = r & DIGIT_MAX;
898  rcarry = r >> DIGIT_SHIFT;
899  }

901  krk_long_trim(res);
902 
903  if (rneg) {
904  krk_long_set_sign(res,-1);
905  }

908 }

1724  static const char _method_name[] = "to_bytes";

1736  CHECK_ARG(2,str,KrkString*,byteorder);
1737  int _signed = 0;
1738  if (hasKw) {
1739  KrkValue tmp;
1740  if (krk_tableGet(AS_DICT(argv[argc]), OBJECT_VAL(S("signed")), &tmp)) {
1741  _signed = !krk_isFalsey(tmp);
1742  }
1743  }
1744 
1745  if (length < 0) {
1746  return krk_runtimeError(vm.exceptions->valueError, "length must be non-negative");
1747  }
1748 
1749  int order = 0;
1750  if (!strcmp(byteorder->chars,"little")) {
1751  order = 1;
1752  } else if (!strcmp(byteorder->chars,"big")) {
1753  order = -1;
1754  } else {
1755  return krk_runtimeError(vm.exceptions->valueError, "byteorder must be either 'little' or 'big'");
1756  }
1757 
1758  if (krk_long_sign(val) < 0 && !_signed) {
1759  return krk_runtimeError(vm.exceptions->notImplementedError, "can not convert negative value to unsigned");
1760  }
1761 
1762  /* We could avoid the copy for a positive value, but whatever... */
1763  KrkLong tmp;
1764  krk_long_init_ui(&tmp, 0);
1765  krk_long_abs(&tmp, val);
1766 
1767  /* Invert negative values; already checked for signed... */
1768  if (krk_long_sign(val) < 0) {
1769  KrkLong one;
1770  krk_long_init_ui(&one, 1);
1771  krk_long_sub(&tmp, &tmp, &one);
1772  krk_long_clear(&one);
1773  }
1774 
1775  /* Use bits from inverted value */
1776  size_t bitCount = _bits_in(&tmp);
1777 
1778  /* If it is signed, we need to reserve the top bit for the sign;
1779  * eg., (127).to_bytes(1,...,signed=True) is fine, but (128) is not,
1780  * and also (-128) should work, which is taken care of by using the
1781  * inverted value... Also, as a weird special case, 0 still has no
1782  * bits even if 'signed', which allows (0).to_bytes(0,...,signed=True)
1783  * even though that makes no sense to me... */
1784  if (_signed && val->width != 0) bitCount++;
1785 
1786  if ((size_t)length * 8 < bitCount) {
1787  krk_long_clear(&tmp);
1788  /* Should be OverflowError, but we don't have that and I don't care to add it right now */
1789  return krk_runtimeError(vm.exceptions->valueError, "int too big to convert");
1790  }
1791 
1792  /* Allocate bytes for final representation */
1793  krk_push(OBJECT_VAL(krk_newBytes(length, NULL)));
1794  memset(AS_BYTES(krk_peek(0))->bytes, 0, length);
1795 
1796  /* We'll use a 'bit reader':
1797  * - We want 8 bits for each byte.
1798  * - We can collect 31 bits from each digit.
1799  * - If we run out of digits, we're done.
1800  */
1801  ssize_t i = 0;
1802  ssize_t j = 0;
1803 
1804  uint64_t accum = 0;
1805  int32_t remaining = 0;
1806  int break_here = 0;
1807 
1808  while (i < length && !break_here) {
1809  if (remaining < 8) {
1810  if (j < tmp.width) {
1811  accum |= ((uint64_t)tmp.digits[j]) << remaining;
1812  j++;
1813  } else {
1814  break_here = 1;
1815  }
1816  remaining += 31;
1817  }
1818 
1819  uint8_t byte = accum & 0xFF;
1820  accum >>= 8;
1821  remaining -= 8;
1822 
1823  AS_BYTES(krk_peek(0))->bytes[order == 1 ? i : (length - i - 1)] = byte;
1824  i++;
1825  }
1826 
1827  /* If input was negative, at this point we're producing an inverted value;
1828  * we already encoded (|n|-1), so now we just need to bit invert. */
1829  if (krk_long_sign(val) < 0) {
1830  for (size_t i = 0; i < (size_t)length; ++i) {
1831  AS_BYTES(krk_peek(0))->bytes[i] ^= 0xFF;
1832  }
1833  }
1834 
1835  /* We produced a bytes object, so we no longer need the long. */
1836  krk_long_clear(&tmp);
1837  return krk_pop();
1838 }

1933 int krk_long_to_int(KrkValue val, char size, void * out) {
1934  uint64_t accum = 0;
1935  if (IS_INTEGER(val)) {
1936  /* For integers, there's nothing to do until we want to start
1937  * doing overflow checking, so just extend to 64-bit. */
1938  accum = AS_INTEGER(val);
1939  } else if (IS_long(val)) {
1940  /* For longs, we have some additional work. */
1941  struct BigInt * self = (void*)AS_OBJECT(val);
1942  KrkLong * this = self->value;
1943  size_t swidth = this->width < 0 ? -this->width : this->width;
1944 
1945  if (swidth > 0) {
1946  /* Collect up to three digits worth of bits, to the maximum
1947  * we support of 64. 31, 31, and 2. */
1948  accum |= (uint64_t)this->digits[0];
1949  if (swidth > 1) {
1950  accum |= (uint64_t)this->digits[1] << DIGIT_SHIFT;
1951  if (swidth > 2) {
1952  accum |= (uint64_t)(this->digits[2] & 0x3) << DIGIT_SHIFT * 2;
1953  }
1954  }
1955  /* If this is a negative value, convert the result to twos-complement. */
1956  if (this->width < 0) {
1957  accum -= 1;
1958  accum ^= 0xFFFFffffFFFFffff;
1959  }
1960  }
1961  } else {
1962  krk_runtimeError(vm.exceptions->typeError, "expected %s, not '%T'", "int", val);
1963  return 0;
1964  }
1965 
1966  /* Now copy over the output. */
1967  switch (size) {
1968  case sizeof(uint8_t): *(uint8_t*)out = accum; break;
1969  case sizeof(uint16_t): *(uint16_t*)out = accum; break;
1970  case sizeof(uint32_t): *(uint32_t*)out = accum; break;
1971  case sizeof(uint64_t): *(uint64_t*)out = accum; break;
1972  default:
1973  krk_runtimeError(vm.exceptions->SystemError, "invalid size");
1974  return 0;
1975  }
1976 
1977  return 1;
1978 }
