66 static KrkValue readLine(char * prompt, int promptWidth, char * syntaxHighlighter) {
67  struct StringBuilder sb = {0};
68 
69 #ifndef NO_RLINE
70  if (enableRline) {
71  rline_exit_string = "";
72  rline_exp_set_prompts(prompt, "", promptWidth, 0);
73  rline_exp_set_syntax(syntaxHighlighter);
74  rline_exp_set_tab_complete_func(NULL);
75  } else
76 #endif
77  {
78  fprintf(stdout, "%s", prompt);
79  fflush(stdout);
80  }
81 
82  /* Read a line of input using a method that we can guarantee will be
83  * interrupted by signal delivery. */
84  while (1) {
85  char buf[4096];
86  ssize_t bytesRead = doRead(buf, 4096);
87  if (krk_currentThread.flags & KRK_THREAD_SIGNALLED) goto _exit;
88  if (bytesRead < 0) {
89  krk_runtimeError(vm.exceptions->ioError, "%s", strerror(errno));
90  goto _exit;
91  } else if (bytesRead == 0 && !sb.length) {
92  krk_runtimeError(vm.exceptions->baseException, "EOF");
93  goto _exit;
94  } else {
95  pushStringBuilderStr(&sb, buf, bytesRead);
96  }
97  /* Was there a linefeed? Then we can exit. */
98  if (sb.length && sb.bytes[sb.length-1] == '\n') {
99  sb.length--;
100  break;
101  }
102  }
103 
104  return finishStringBuilder(&sb);
105 
106 _exit:
107  discardStringBuilder(&sb);
108  return NONE_VAL();
109 }

179 static void tab_complete_func(rline_context_t * c) {
180  /* Figure out where the cursor is and if we should be completing anything. */
181  if (c->offset) {
182  /* Copy up to the cursor... */
183  char * tmp = malloc(c->offset + 1);
184  memcpy(tmp, c->buffer, c->offset);
185  tmp[c->offset] = '\0';
186  /* and pass it to the scanner... */
187  krk_initScanner(tmp);
188  /* Logically, there can be at most (offset) tokens, plus some wiggle room. */
189  KrkToken * space = malloc(sizeof(KrkToken) * (c->offset + 2));
190  int count = 0;
191  do {
192  space[count++] = krk_scanToken();
193  } while (space[count-1].type != TOKEN_EOF && space[count-1].type != TOKEN_ERROR);
194 
195  /* If count == 1, it was EOF or an error and we have nothing to complete. */
196  if (count == 1) {
197  goto _cleanup;
198  }
199 
200  /* Otherwise we want to see if we're on an identifier or a dot. */
201  int base = 2;
202  int n = base;
203  if (space[count-base].type == TOKEN_DOT) {
204  /* Dots we need to look back at the previous tokens for */
205  n--;
206  base--;
207  } else if (space[count-base].type >= TOKEN_IDENTIFIER && space[count-base].type <= TOKEN_WITH) {
208  /* Something alphanumeric; only for the last element */
209  } else {
210  /* Some other symbol */
211  goto _cleanup;
212  }
213 
214  /* Work backwards to find the start of this chain of identifiers */
215  while (n < count) {
216  if (space[count-n-1].type != TOKEN_DOT) break;
217  n++;
218  if (n == count) break;
219  if (space[count-n-1].type != TOKEN_IDENTIFIER) break;
220  n++;
221  }
222 
223  if (n <= count) {
224  /* Now work forwards, starting from the current globals. */
225  KrkValue root = OBJECT_VAL(krk_currentThread.module);
226  int isGlobal = 1;
227  while (n > base) {
228  /* And look at the potential fields for instances/classes */
229  KrkValue next = findFromProperty(root, space[count-n]);
230  if (IS_NONE(next)) {
231  /* If we hit None, we found something invalid (or literally hit a None
232  * object, but really the difference is minimal in this case: Nothing
233  * useful to tab complete from here. */
234  goto _cleanup;
235  }
236  isGlobal = 0;
237  root = next;
238  n -= 2; /* To skip every other dot. */
239  }
240 
241  /* Now figure out what we're completing - did we already have a partial symbol name? */
242  int length = (space[count-base].type == TOKEN_DOT) ? 0 : (space[count-base].length);
243  isGlobal = isGlobal && (length != 0);
244 
245  /* Collect up to 256 of those that match */
246  char * matches[256];
247  int matchCount = 0;
248 
249  /* Take the last symbol name from the chain and get its member list from dir() */
250 
251  for (;;) {
252  KrkValue dirList = krk_dirObject(1,(KrkValue[]){root},0);
253  krk_push(dirList);
254  if (!IS_INSTANCE(dirList)) {
255  fprintf(stderr,"\nInternal error while tab completting.\n");
256  goto _cleanup;
257  }
258 
259  for (size_t i = 0; i < AS_LIST(dirList)->count; ++i) {
260  KrkString * s = AS_STRING(AS_LIST(dirList)->values[i]);
261  krk_push(OBJECT_VAL(s));
262  KrkToken asToken = {.start = s->chars, .literalWidth = s->length};
263  KrkValue thisValue = findFromProperty(root, asToken);
264  krk_push(thisValue);
265  if (IS_CLOSURE(thisValue) || IS_BOUND_METHOD(thisValue) ||
266  (IS_NATIVE(thisValue) && ((KrkNative*)AS_OBJECT(thisValue))->isMethod != 2)) {
267  size_t allocSize = s->length + 2;
268  char * tmp = malloc(allocSize);
269  size_t len = snprintf(tmp, allocSize, "%s(", s->chars);
270  s = krk_takeString(tmp, len);
271  krk_pop();
272  krk_push(OBJECT_VAL(s));
273  }
274 
275  /* If this symbol is shorter than the current submatch, skip it. */
276  if (length && (int)s->length < length) continue;
277 
278  /* See if it's already in the matches */
279  int found = 0;
280  for (int i = 0; i < matchCount; ++i) {
281  if (!strcmp(matches[i], s->chars)) {
282  found = 1;
283  break;
284  }
285  }
286  if (found) continue;
287 
288  if (!memcmp(s->chars, space[count-base].start, length)) {
289  matches[matchCount] = s->chars;
290  matchCount++;
291  if (matchCount == 255) goto _toomany;
292  }
293  }
294 
295  /*
296  * If the object we were scanning was the current module,
297  * then we should also throw the builtins into the ring.
298  */
299  if (isGlobal && AS_OBJECT(root) == (KrkObj*)krk_currentThread.module) {
300  root = OBJECT_VAL(vm.builtins);
301  continue;
302  } else if (isGlobal && AS_OBJECT(root) == (KrkObj*)vm.builtins) {
303  extern char * syn_krk_keywords[];
304  KrkInstance * fakeKeywordsObject = krk_newInstance(vm.baseClasses->objectClass);
305  root = OBJECT_VAL(fakeKeywordsObject);
306  krk_push(root);
307  for (char ** keyword = syn_krk_keywords; *keyword; keyword++) {
308  krk_attachNamedValue(&fakeKeywordsObject->fields, *keyword, NONE_VAL());
309  }
310  continue;
311  } else {
312  break;
313  }
314  }
315 _toomany:
316 
317  /* Now we can do things with the matches. */
318  if (matchCount == 1) {
319  /* If there was only one, just fill it. */
320  rline_insert(c, matches[0] + length);
321  rline_place_cursor();
322  } else if (matchCount) {
323  /* Otherwise, try to find a common substring among them... */
324  int j = length;
325  while (1) {
326  char m = matches[0][j];
327  if (!m) break;
328  int diff = 0;
329  for (int i = 1; i < matchCount; ++i) {
330  if (matches[i][j] != m) {
331  diff = 1;
332  break;
333  }
334  }
335  if (diff) break;
336  j++;
337  }
338  /* If no common sub string could be filled in, we print the list. */
339  if (j == length) {
340  /* First find the maximum width of an entry */
341  int maxWidth = 0;
342  for (int i = 0; i < matchCount; ++i) {
343  if ((int)strlen(matches[i]) > maxWidth) maxWidth = strlen(matches[i]);
344  }
345  /* Now how many can we fit in a screen */
346  int colsPerLine = rline_terminal_width / (maxWidth + 2); /* +2 for the spaces */
347  fprintf(stderr, "\n");
348  int column = 0;
349  for (int i = 0; i < matchCount; ++i) {
350  fprintf(stderr, "%-*s ", maxWidth, matches[i]);
351  column += 1;
352  if (column >= colsPerLine) {
353  fprintf(stderr, "\n");
354  column = 0;
355  }
356  }
357  if (column != 0) fprintf(stderr, "\n");
358  } else {
359  /* If we do have a common sub string, fill in those characters. */
360  for (int i = length; i < j; ++i) {
361  char tmp[2] = {matches[0][i], '\0'};
362  rline_insert(c, tmp);
363  }
364  }
365  }
366  }
367 _cleanup:
368  free(tmp);
369  free(space);
370  krk_resetStack();
371  return;
372  }
373 }

640 static void findInterpreter(char * argv[]) {
641 #ifdef _WIN32
642  vm.binpath = strdup(_pgmptr);
643 #else
644  /* Try asking /proc */
645  char tmp[4096];
646  char * binpath = realpath("/proc/self/exe", tmp);
647  if (!binpath || (access(binpath, X_OK) != 0)) {
648  if (strchr(argv[0], '/')) {
649  binpath = realpath(argv[0], tmp);
650  } else {
651  /* Search PATH for argv[0] */
652  char * _path = strdup(getenv("PATH"));
653  char * path = _path;
654  while (path) {
655  char * next = strchr(path,':');
656  if (next) *next++ = '\0';
657 
658  snprintf(tmp, 4096, "%s/%s", path, argv[0]);
659  if (access(tmp, X_OK) == 0) {
660  binpath = tmp;
661  break;
662  }
663  path = next;
664  }
665  free(_path);
666  }
667  }
668  if (binpath) {
669  vm.binpath = strdup(binpath);
670  } /* Else, give up at this point and just don't attach it at all. */
