67 static KrkValue readLine(char * prompt, int promptWidth, char * syntaxHighlighter) {
68  struct StringBuilder sb = {0};
69 
70 #ifndef NO_RLINE
71  if (enableRline) {
72  rline_exit_string = "";
73  rline_exp_set_prompts(prompt, "", promptWidth, 0);
74  rline_exp_set_syntax(syntaxHighlighter);
75  rline_exp_set_tab_complete_func(NULL);
76  } else
77 #endif
78  {
79  fprintf(stdout, "%s", prompt);
80  fflush(stdout);
81  }
82 
83  /* Read a line of input using a method that we can guarantee will be
84  * interrupted by signal delivery. */
85  while (1) {
86  char buf[4096];
87  ssize_t bytesRead = doRead(buf, 4096);
88  if (krk_currentThread.flags & KRK_THREAD_SIGNALLED) goto _exit;
89  if (bytesRead < 0) {
90  krk_runtimeError(vm.exceptions->ioError, "%s", strerror(errno));
91  goto _exit;
92  } else if (bytesRead == 0 && !sb.length) {
93  krk_runtimeError(vm.exceptions->baseException, "EOF");
94  goto _exit;
95  } else {
96  pushStringBuilderStr(&sb, buf, bytesRead);
97  }
98  /* Was there a linefeed? Then we can exit. */
99  if (sb.length && sb.bytes[sb.length-1] == '\n') {
100  sb.length--;
101  break;
102  }
103  }
104 
105  return finishStringBuilder(&sb);
106 
107 _exit:
108  discardStringBuilder(&sb);
109  return NONE_VAL();
110 }

165 static void tab_complete_func(rline_context_t * c) {
166  /* Figure out where the cursor is and if we should be completing anything. */
167  if (c->offset) {
168  /* Copy up to the cursor... */
169  char * tmp = malloc(c->offset + 1);
170  memcpy(tmp, c->buffer, c->offset);
171  tmp[c->offset] = '\0';
172  /* and pass it to the scanner... */
173  krk_initScanner(tmp);
174  /* Logically, there can be at most (offset) tokens, plus some wiggle room. */
175  KrkToken * space = malloc(sizeof(KrkToken) * (c->offset + 2));
176  int count = 0;
177  do {
178  space[count++] = krk_scanToken();
179  } while (space[count-1].type != TOKEN_EOF && space[count-1].type != TOKEN_ERROR);
180 
181  /* If count == 1, it was EOF or an error and we have nothing to complete. */
182  if (count == 1) {
183  goto _cleanup;
184  }
185 
186  /* Otherwise we want to see if we're on an identifier or a dot. */
187  int base = 2;
188  int n = base;
189  if (space[count-base].type == TOKEN_DOT) {
190  /* Dots we need to look back at the previous tokens for */
191  n--;
192  base--;
193  } else if (space[count-base].type >= TOKEN_IDENTIFIER && space[count-base].type <= TOKEN_WITH) {
194  /* Something alphanumeric; only for the last element */
195  } else {
196  /* Some other symbol */
197  goto _cleanup;
198  }
199 
200  /* Work backwards to find the start of this chain of identifiers */
201  while (n < count) {
202  if (space[count-n-1].type != TOKEN_DOT) break;
203  n++;
204  if (n == count) break;
205  if (space[count-n-1].type != TOKEN_IDENTIFIER) break;
206  n++;
207  }
208 
209  if (n <= count) {
210  /* Now work forwards, starting from the current globals. */
211  KrkValue root = OBJECT_VAL(krk_currentThread.module);
212  int isGlobal = 1;
213  while (n > base) {
214  /* And look at the potential fields for instances/classes */
215  KrkValue next = findFromProperty(root, space[count-n]);
216  if (IS_NONE(next)) {
217  /* If we hit None, we found something invalid (or literally hit a None
218  * object, but really the difference is minimal in this case: Nothing
219  * useful to tab complete from here. */
220  if (!isGlobal) goto _cleanup;
221  /* Does this match a builtin? */
222  if (!krk_tableGet_fast(&vm.builtins->fields,
223  krk_copyString(space[count-n].start,space[count-n].literalWidth), &next) || IS_NONE(next)) {
224  goto _cleanup;
225  }
226  }
227  isGlobal = 0;
228  root = next;
229  n -= 2; /* To skip every other dot. */
230  }
231 
232  /* Now figure out what we're completing - did we already have a partial symbol name? */
233  int length = (space[count-base].type == TOKEN_DOT) ? 0 : (space[count-base].length);
234  isGlobal = isGlobal && (length != 0);
235 
236  /* Collect up to 256 of those that match */
237  char * matches[256];
238  int matchCount = 0;
239 
240  /* Take the last symbol name from the chain and get its member list from dir() */
241 
242  for (;;) {
243  KrkValue dirList = krk_dirObject(1,(KrkValue[]){root},0);
244  krk_push(dirList);
245  if (!IS_INSTANCE(dirList)) {
246  fprintf(stderr,"\nInternal error while tab completting.\n");
247  goto _cleanup;
248  }
249 
250  for (size_t i = 0; i < AS_LIST(dirList)->count; ++i) {
251  KrkString * s = AS_STRING(AS_LIST(dirList)->values[i]);
252  krk_push(OBJECT_VAL(s));
253  KrkToken asToken = {.start = s->chars, .literalWidth = s->length};
254  KrkValue thisValue = findFromProperty(root, asToken);
255  krk_push(thisValue);
256  if (IS_CLOSURE(thisValue) || IS_BOUND_METHOD(thisValue) ||
257  (IS_NATIVE(thisValue) && !(((KrkNative*)AS_OBJECT(thisValue))->flags & KRK_NATIVE_FLAGS_IS_DYNAMIC_PROPERTY))) {
258  size_t allocSize = s->length + 2;
259  char * tmp = malloc(allocSize);
260  size_t len = snprintf(tmp, allocSize, "%s(", s->chars);
261  s = krk_takeString(tmp, len);
262  krk_pop();
263  krk_push(OBJECT_VAL(s));
264  }
265 
266  /* If this symbol is shorter than the current submatch, skip it. */
267  if (length && (int)s->length < length) continue;
268 
269  /* See if it's already in the matches */
270  int found = 0;
271  for (int i = 0; i < matchCount; ++i) {
272  if (!strcmp(matches[i], s->chars)) {
273  found = 1;
274  break;
275  }
276  }
277  if (found) continue;
278 
279  if (!memcmp(s->chars, space[count-base].start, length)) {
280  matches[matchCount] = s->chars;
281  matchCount++;
282  if (matchCount == 255) goto _toomany;
283  }
284  }
285 
286  /*
287  * If the object we were scanning was the current module,
288  * then we should also throw the builtins into the ring.
289  */
290  if (isGlobal && AS_OBJECT(root) == (KrkObj*)krk_currentThread.module) {
291  root = OBJECT_VAL(vm.builtins);
292  continue;
293  } else if (isGlobal && AS_OBJECT(root) == (KrkObj*)vm.builtins) {
294  extern char * syn_krk_keywords[];
295  KrkInstance * fakeKeywordsObject = krk_newInstance(vm.baseClasses->objectClass);
296  root = OBJECT_VAL(fakeKeywordsObject);
297  krk_push(root);
298  for (char ** keyword = syn_krk_keywords; *keyword; keyword++) {
299  krk_attachNamedValue(&fakeKeywordsObject->fields, *keyword, NONE_VAL());
300  }
301  continue;
302  } else {
303  break;
304  }
305  }
306 _toomany:
307 
308  /* Now we can do things with the matches. */
309  if (matchCount == 1) {
310  /* If there was only one, just fill it. */
311  rline_insert(c, matches[0] + length);
312  rline_place_cursor();
313  } else if (matchCount) {
314  /* Otherwise, try to find a common substring among them... */
315  int j = length;
316  while (1) {
317  char m = matches[0][j];
318  if (!m) break;
319  int diff = 0;
320  for (int i = 1; i < matchCount; ++i) {
321  if (matches[i][j] != m) {
322  diff = 1;
323  break;
324  }
325  }
326  if (diff) break;
327  j++;
328  }
329  /* If no common sub string could be filled in, we print the list. */
330  if (j == length) {
331  /* First find the maximum width of an entry */
332  int maxWidth = 0;
333  for (int i = 0; i < matchCount; ++i) {
334  if ((int)strlen(matches[i]) > maxWidth) maxWidth = strlen(matches[i]);
335  }
336  /* Now how many can we fit in a screen */
337  int colsPerLine = rline_terminal_width / (maxWidth + 2); /* +2 for the spaces */
338  fprintf(stderr, "\n");
339  int column = 0;
340  for (int i = 0; i < matchCount; ++i) {
341  fprintf(stderr, "%-*s ", maxWidth, matches[i]);
342  column += 1;
343  if (column >= colsPerLine) {
344  fprintf(stderr, "\n");
345  column = 0;
346  }
347  }
348  if (column != 0) fprintf(stderr, "\n");
349  } else {
350  /* If we do have a common sub string, fill in those characters. */
351  for (int i = length; i < j; ++i) {
352  char tmp[2] = {matches[0][i], '\0'};
353  rline_insert(c, tmp);
354  }
355  }
356  }
357  }
358 _cleanup:
359  free(tmp);
360  free(space);
361  krk_resetStack();
362  return;
363  }
364 }

659 static void findInterpreter(char * argv[]) {
660 #ifdef _WIN32
661  vm.binpath = strdup(_pgmptr);
662 #else
663  /* Try asking /proc */
664  char tmp[4096];
665  char * binpath = realpath("/proc/self/exe", tmp);
666  if (!binpath || (access(binpath, X_OK) != 0)) {
667  if (strchr(argv[0], '/')) {
668  binpath = realpath(argv[0], tmp);
669  } else {
670  /* Search PATH for argv[0] */
671  char * p = getenv("PATH");
672  if (!p) return;
673  char * _path = strdup(p);
674  char * path = _path;
675  while (path) {
676  char * next = strchr(path,':');
677  if (next) *next++ = '\0';
