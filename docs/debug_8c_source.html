141 size_t krk_disassembleInstruction(FILE * f, KrkFunction * func, size_t offset) {
142  KrkChunk * chunk = &func->chunk;
143  if (offset > 0 && krk_lineNumber(chunk, offset) == krk_lineNumber(chunk, offset - 1)) {
144  fprintf(f, " ");
145  } else {
146  if (offset > 0) fprintf(f,"\n");
147  fprintf(f, "%4d ", (int)krk_lineNumber(chunk, offset));
148  }
149  if (isJumpTarget(func,offset)) {
150  fprintf(f, " >> ");
151  } else {
152  fprintf(f, " ");
153  }
154  fprintf(f, "%4u ", (unsigned int)offset);
155  uint8_t opcode = chunk->code[offset];
156  size_t size = 1;
157 
158  switch (opcode) {
159 #include "opcodes.h"
160  default:
161  fprintf(f, "Unknown opcode: %02x", opcode);
162  }
163 
164  /* Birthdays - Local names that have become valid from this instruction */
165  for (size_t i = 0; i < func->localNameCount; ++i) {
166  if (func->localNames[i].birthday >= offset && func->localNames[i].birthday < offset + size) {
167  fprintf(f, " +%s", func->localNames[i].name->chars);
168  }
169  }
170 
171  /* Deathdays - Local names that are no longer valid as of this instruction */
172  for (size_t i = 0; i < func->localNameCount; ++i) {
173  if (func->localNames[i].deathday >= offset && func->localNames[i].deathday < offset + size) {
174  fprintf(f, " -%s", func->localNames[i].name->chars);
175  }
176  }
177 
178  fprintf(f,"\n");
179 
180  return offset + size;
181 }
