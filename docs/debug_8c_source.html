248 size_t krk_disassembleInstruction(FILE * f, KrkCodeObject * func, size_t offset) {
249  KrkChunk * chunk = &func->chunk;
250  if (offset > 0 && krk_lineNumber(chunk, offset) == krk_lineNumber(chunk, offset - 1)) {
251  fprintf(f, " ");
252  } else {
253  if (offset > 0) fprintf(f,"\n");
254  fprintf(f, "%4d ", (int)krk_lineNumber(chunk, offset));
255  }
256  if (isJumpTarget(func,offset)) {
257  fprintf(f, " >> ");
258  } else {
259  fprintf(f, " ");
260  }
261  fprintf(f, "%4u ", (unsigned int)offset);
262  uint8_t opcode = chunk->code[offset];
263  size_t size = 1;
264 
265  switch (opcode) {
266 #include "opcodes.h"
267  default:
268  fprintf(f, "Unknown opcode: %02x", opcode);
269  }
270 
271  /* Birthdays - Local names that have become valid from this instruction */
272  for (size_t i = 0; i < func->localNameCount; ++i) {
273  if (func->localNames[i].birthday >= offset && func->localNames[i].birthday < offset + size) {
274  fprintf(f, " +%s", func->localNames[i].name->chars);
275  }
276  }
277 
278  /* Deathdays - Local names that are no longer valid as of this instruction */
279  for (size_t i = 0; i < func->localNameCount; ++i) {
280  if (func->localNames[i].deathday >= offset && func->localNames[i].deathday < offset + size) {
281  fprintf(f, " -%s", func->localNames[i].name->chars);
282  }
283  }
284 
285  fprintf(f,"\n");
286 
287  return offset + size;
288 }

327 
328  KrkCodeObject * target = NULL;
329 
330  /* Examine all code objects to find one that matches the requested
331  * filename and line number... */
332  KrkObj * object = vm.objects;
333  while (object) {
334  if (object->type == KRK_OBJ_CODEOBJECT) {
335  KrkChunk * chunk = &((KrkCodeObject*)object)->chunk;
336  if (filename == chunk->filename) {
337  /* We have a candidate. */
338  if (krk_lineNumber(chunk, 0) <= line &&
339  krk_lineNumber(chunk,chunk->count) >= line) {
340  target = (KrkCodeObject*)object;
341  break;
342  }
343  }
344  }
345  object = object->next;
346  }
347 
348  /* No matching function was found... */
349  if (!target) return -1;

352

354  for (size_t i = 0; i < target->chunk.linesCount; ++i) {
355  if (target->chunk.lines[i].line > line) break;
356  if (target->chunk.lines[i].line == line) {
357  offset = target->chunk.lines[i].startOffset;
358  break;

360  offset = target->chunk.lines[i].startOffset;
361  }

428 int krk_debuggerHook(KrkCallFrame * frame) {
429  if (!vm.dbgState->debuggerHook)
430  abort();
431 
432  if (vm.dbgState->repeatStack_top != -1) {
433  /* Re-enable stored repeat breakpoint */
434  krk_debug_enableBreakpoint(vm.dbgState->repeatStack_top);
435  }
436 
437  vm.dbgState->repeatStack_top = vm.dbgState->repeatStack_bottom;
438  vm.dbgState->repeatStack_bottom = -1;
439 
440  if (!vm.dbgState->thisWasForced) {
441  int result = vm.dbgState->debuggerHook(frame);
442  switch (result) {
443  case KRK_DEBUGGER_CONTINUE:
444  krk_debug_disableSingleStep();
445  break;
446  case KRK_DEBUGGER_ABORT:
447  abort();
448  break;
449  case KRK_DEBUGGER_STEP:
450  krk_debug_enableSingleStep();
451  break;
452  case KRK_DEBUGGER_QUIT:
453  exit(0);
454  break;
455  case KRK_DEBUGGER_RAISE:
456  krk_runtimeError(vm.exceptions->baseException, "raise from debugger");
457  break;
458  }
459  } else {
460  /* If we weren't asked to step to the next breakpoint, we need to disable single stepping. */
461  krk_debug_disableSingleStep();
462  vm.dbgState->thisWasForced = 0;
463  }
464 
465  /* If the top of the repeat stack is an index, we need to ensure we re-enable that breakpoint */
466  if (vm.dbgState->repeatStack_top != -1 && !(krk_currentThread.flags & KRK_THREAD_SINGLE_STEP)) {
467  vm.dbgState->thisWasForced = 1;
468  krk_debug_enableSingleStep();
469  }
470 
471  return 0;
472 }

494 int krk_debugBreakpointHandler(void) {
495  int index = -1;
496 
497  KrkCallFrame * frame = &krk_currentThread.frames[krk_currentThread.frameCount-1];
498  KrkCodeObject * callee = frame->closure->function;
499  size_t offset = (frame->ip - 1) - callee->chunk.code;
500 
501  for (int i = 0; i < vm.dbgState->breakpointsCount; ++i) {
502  if (vm.dbgState->breakpoints[i].inFunction == callee && vm.dbgState->breakpoints[i].offset == offset) {
503  index = i;
504  }
505  }
506 
507  /* A breakpoint instruction without an associated breakpoint entry?
508  * This must be an invalid block of bytecode - just bail! */
509  if (index == -1) {
510  abort();
511  }
512 
513  /* Restore the instruction to its original state. If the debugger
514  * wants to break here again it can set the breakpoint again */
515  callee->chunk.code[offset] = vm.dbgState->breakpoints[index].originalOpcode;
516 
517  /* If this was a single-shot, we can remove it. */
518  if (vm.dbgState->breakpoints[index].flags == KRK_BREAKPOINT_ONCE) {
519  krk_debug_removeBreakpoint(index);
520  } else if (vm.dbgState->breakpoints[index].flags == KRK_BREAKPOINT_REPEAT) {
521  vm.dbgState->repeatStack_bottom = index;
522  }
523 
524  /* Rewind to rerun this instruction. */
525  frame->ip--;
526 
527  return krk_debuggerHook(frame);
528 }
