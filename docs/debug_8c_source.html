18 void krk_debug_dumpStack(FILE * file, KrkCallFrame * frame) {
19  size_t i = 0;
20  if (!frame) frame = &krk_currentThread.frames[krk_currentThread.frameCount-1];
21  for (KrkValue * slot = krk_currentThread.stack; slot < krk_currentThread.stackTop; slot++) {
22  fprintf(file, "[%c", frame->slots == i ? '*' : ' ');
23 
24  for (size_t x = krk_currentThread.frameCount; x > 0; x--) {
25  if (krk_currentThread.frames[x-1].slots > i) continue;
26  KrkCallFrame * f = &krk_currentThread.frames[x-1];
27  size_t relative = i - f->slots;
28 
29  /* Figure out the name of this value */
30  if (relative < (size_t)f->closure->function->requiredArgs) {
31  fprintf(file, "%s=", AS_CSTRING(f->closure->function->requiredArgNames.values[relative]));
32  break;
33  } else if (relative < (size_t)f->closure->function->requiredArgs + (size_t)f->closure->function->keywordArgs) {
34  fprintf(file, "%s=", AS_CSTRING(f->closure->function->keywordArgNames.values[relative - f->closure->function->requiredArgs]));
35  break;
36  } else {
37  int found = 0;
38  for (size_t j = 0; j < f->closure->function->localNameCount; ++j) {
39  if (relative == f->closure->function->localNames[j].id
40  /* Only display this name if it's currently valid */
41  && f->closure->function->localNames[j].birthday <= (size_t)(f->ip - f->closure->function->chunk.code)
42  && f->closure->function->localNames[j].deathday >= (size_t)(f->ip - f->closure->function->chunk.code)
43  ) {
44  fprintf(file, "%s=", f->closure->function->localNames[j].name->chars);
45  found = 1;
46  break;
47  }
48  }
49  if (found) break;
50  }
51  }
52 
53  krk_printValueSafe(file, *slot);
54  fprintf(file, " ]");
55  i++;
56  }
57  if (i == frame->slots) {
58  fprintf(file, " * ");
59  }
60  fprintf(file, "\n");
61 }

188 size_t krk_disassembleInstruction(FILE * f, KrkCodeObject * func, size_t offset) {
189  KrkChunk * chunk = &func->chunk;
190  if (offset > 0 && krk_lineNumber(chunk, offset) == krk_lineNumber(chunk, offset - 1)) {
191  fprintf(f, " ");
192  } else {
193  if (offset > 0) fprintf(f,"\n");
194  fprintf(f, "%4d ", (int)krk_lineNumber(chunk, offset));
195  }
196  if (isJumpTarget(func,offset)) {
197  fprintf(f, " >> ");
198  } else {
199  fprintf(f, " ");
200  }
201  fprintf(f, "%4u ", (unsigned int)offset);
202  uint8_t opcode = chunk->code[offset];
203  size_t size = 1;
204 
205  switch (opcode) {
206 #include "opcodes.h"
207  default:
208  fprintf(f, "Unknown opcode: %02x", opcode);
209  }
210 
211  /* Birthdays - Local names that have become valid from this instruction */
212  for (size_t i = 0; i < func->localNameCount; ++i) {
213  if (func->localNames[i].birthday >= offset && func->localNames[i].birthday < offset + size) {
214  fprintf(f, " +%s", func->localNames[i].name->chars);
215  }
216  }
217 
218  /* Deathdays - Local names that are no longer valid as of this instruction */
219  for (size_t i = 0; i < func->localNameCount; ++i) {
220  if (func->localNames[i].deathday >= offset && func->localNames[i].deathday < offset + size) {
221  fprintf(f, " -%s", func->localNames[i].name->chars);
222  }
223  }
224 
225  fprintf(f,"\n");
226 
227  return offset + size;
228 }

262 int krk_debug_addBreakpointCodeOffset(KrkCodeObject * target, size_t offset, int flags) {
263  int index = breakpointsCount;
264  if (breakpointsCount == MAX_BREAKPOINTS) {
265  /* See if any are available */
266  for (int i = 0; i < MAX_BREAKPOINTS; ++i) {
267  if (breakpoints[i].inFunction == NULL) {
268  index = i;
269  break;
270  }
271  }
272  if (index == breakpointsCount) {
273  return -1;
274  }
275  } else {
276  index = breakpointsCount++;
277  }
278 
279  breakpoints[index].inFunction = target;
280  breakpoints[index].offset = offset;
281  breakpoints[index].originalOpcode = target->chunk.code[offset];
282  breakpoints[index].flags = flags;
283  target->chunk.code[offset] = OP_BREAKPOINT;
284 
285  return index;
286 }

289 
290  KrkCodeObject * target = NULL;
291 
292  /* Examine all code objects to find one that matches the requested
293  * filename and line number... */
294  KrkObj * object = vm.objects;
295  while (object) {
296  if (object->type == KRK_OBJ_CODEOBJECT) {
297  KrkChunk * chunk = &((KrkCodeObject*)object)->chunk;
298  if (filename == chunk->filename) {
299  /* We have a candidate. */
300  if (krk_lineNumber(chunk, 0) <= line &&
301  krk_lineNumber(chunk,chunk->count) >= line) {
302  target = (KrkCodeObject*)object;
303  break;
304  }
305  }
306  }
307  object = object->next;
308  }
309 
310  /* No matching function was found... */
311  if (!target) return -1;
312 
313  /* Find the right offset in this function */
314 
315  size_t offset = 0;
316  for (size_t i = 0; i < target->chunk.linesCount; ++i) {
317  if (target->chunk.lines[i].line > line) break;
318  if (target->chunk.lines[i].line == line) {
319  offset = target->chunk.lines[i].startOffset;
320  break;
321  }
322  offset = target->chunk.lines[i].startOffset;
323  }
324

454 int krk_debuggerHook(KrkCallFrame * frame) {
455  if (!_debugger_hook)
456  abort();
457 
458  if (_repeatStack_top != -1) {
459  /* Re-enable stored repeat breakpoint */
460  krk_debug_enableBreakpoint(_repeatStack_top);
461  }
462 
463  _repeatStack_top = _repeatStack_bottom;
464  _repeatStack_bottom = -1;
465 
466  if (!_thisWasForced) {
467  int result = _debugger_hook(frame);
468  switch (result) {
469  case KRK_DEBUGGER_CONTINUE:
470  krk_debug_disableSingleStep();
471  break;
472  case KRK_DEBUGGER_ABORT:
473  abort();
474  break;
475  case KRK_DEBUGGER_STEP:
476  krk_debug_enableSingleStep();
477  break;
478  case KRK_DEBUGGER_QUIT:
479  exit(0);
480  break;
481  case KRK_DEBUGGER_RAISE:
482  krk_runtimeError(vm.exceptions->baseException, "raise from debugger");
483  break;
484  }
485  } else {
486  /* If we weren't asked to step to the next breakpoint, we need to disable single stepping. */
487  krk_debug_disableSingleStep();
488  _thisWasForced = 0;
489  }
490 
491  /* If the top of the repeat stack is an index, we need to ensure we re-enable that breakpoint */
492  if (_repeatStack_top != -1 && !(krk_currentThread.flags & KRK_THREAD_SINGLE_STEP)) {
493  _thisWasForced = 1;
494  krk_debug_enableSingleStep();
495  }
496 
497  return 0;
498 }

520 int krk_debugBreakpointHandler(void) {
521  int index = -1;
522 
523  KrkCallFrame * frame = &krk_currentThread.frames[krk_currentThread.frameCount-1];
524  KrkCodeObject * callee = frame->closure->function;
525  size_t offset = (frame->ip - 1) - callee->chunk.code;
526 
527  for (int i = 0; i < breakpointsCount; ++i) {
528  if (breakpoints[i].inFunction == callee && breakpoints[i].offset == offset) {
529  index = i;
530  }
531  }
532 
533  /* A breakpoint instruction without an associated breakpoint entry?
534  * This must be an invalid block of bytecode - just bail! */
535  if (index == -1) {
536  abort();
537  }
538 
539  /* Restore the instruction to its original state. If the debugger
540  * wants to break here again it can set the breakpoint again */
541  callee->chunk.code[offset] = breakpoints[index].originalOpcode;
542 
543  /* If this was a single-shot, we can remove it. */
544  if (breakpoints[index].flags == KRK_BREAKPOINT_ONCE) {
545  krk_debug_removeBreakpoint(index);
546  } else if (breakpoints[index].flags == KRK_BREAKPOINT_REPEAT) {
547  _repeatStack_bottom = index;
548  }
549 
550  /* Rewind to rerun this instruction. */
551  frame->ip--;
552 
553  return krk_debuggerHook(frame);
554 }

627 static KrkValue _examineInternal(KrkCodeObject* func) {
628  KrkValue output = krk_list_of(0,NULL,0);
629  krk_push(output);
630 
631  KrkChunk * chunk = &func->chunk;
632  size_t offset = 0;
633  while (offset < chunk->count) {
634  uint8_t opcode = chunk->code[offset];
635  size_t size = 0;
636  ssize_t constant = -1;
637  ssize_t jump = 0;
638  ssize_t operand = -1;
639  ssize_t local = -1;
640  switch (opcode) {
641 #include "opcodes.h"
642  }
643 
644  KrkTuple * newTuple = krk_newTuple(3);
645  krk_push(OBJECT_VAL(newTuple));
646  newTuple->values.values[newTuple->values.count++] = INTEGER_VAL(opcode);
647  newTuple->values.values[newTuple->values.count++] = INTEGER_VAL(size);
648  if (constant != -1) {
649  newTuple->values.values[newTuple->values.count++] = chunk->constants.values[constant];
650  } else if (jump != 0) {
651  newTuple->values.values[newTuple->values.count++] = INTEGER_VAL(jump);
652  } else if (local != -1) {
653  if ((short int)local < func->requiredArgs) {
654  newTuple->values.values[newTuple->values.count++] = func->requiredArgNames.values[local];
655  } else if ((short int)local < func->requiredArgs + func->keywordArgs) {
656  newTuple->values.values[newTuple->values.count++] = func->keywordArgNames.values[local - func->requiredArgs];
657  } else {
658  newTuple->values.values[newTuple->values.count++] = INTEGER_VAL(operand); /* Just in case */
659  for (size_t i = 0; i < func->localNameCount; ++i) {
660  if (func->localNames[i].id == (size_t)local && func->localNames[i].birthday <= offset && func->localNames[i].deathday >= offset) {
661  newTuple->values.values[newTuple->values.count-1] = OBJECT_VAL(func->localNames[i].name);
662  break;
663  }
664  }
665  }
666  } else if (operand != -1) {
667  newTuple->values.values[newTuple->values.count++] = INTEGER_VAL(operand);
668  } else {
669  newTuple->values.values[newTuple->values.count++] = NONE_VAL();
670  }
671  krk_writeValueArray(AS_LIST(output), krk_peek(0));
672  krk_pop();
673 
674  if (size == 0) {
675  abort();
676  }
677 
678  offset += size;
679  }
680 
681  return krk_pop();
682 }
