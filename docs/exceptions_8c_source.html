240  dumpInnerException(inner, depth + 1);

242  } else if (krk_tableGet(&AS_INSTANCE(exception)->fields, OBJECT_VAL(S("__context__")), &inner) && !IS_NONE(inner)) {
243  dumpInnerException(inner, depth + 1);
244  fprintf(stderr, "\nDuring handling of the above exception, another exception occurred:\n\n");
245  }

247  KrkValue tracebackEntries;
248  if (krk_tableGet(&AS_INSTANCE(exception)->fields, OBJECT_VAL(S("traceback")), &tracebackEntries)
249  && IS_list(tracebackEntries) && AS_LIST(tracebackEntries)->count > 0) {
250 
251  /* This exception has a traceback we can print. */
252  fprintf(stderr, "Traceback (most recent call last):\n");
253  for (size_t i = 0; i < AS_LIST(tracebackEntries)->count; ++i) {
254 
255  /* Quietly skip invalid entries as we don't want to bother printing explanatory text for them */
256  if (!IS_TUPLE(AS_LIST(tracebackEntries)->values[i])) continue;
257  KrkTuple * entry = AS_TUPLE(AS_LIST(tracebackEntries)->values[i]);
258  if (entry->values.count != 2) continue;
259  if (!IS_CLOSURE(entry->values.values[0])) continue;
260  if (!IS_INTEGER(entry->values.values[1])) continue;
261 
262  /* Get the function and instruction index from this traceback entry */
263  KrkClosure * closure = AS_CLOSURE(entry->values.values[0]);
264  KrkCodeObject * function = closure->function;
265  size_t instruction = AS_INTEGER(entry->values.values[1]);
266 
267  /* Calculate the line number */
268  int lineNo = (int)krk_lineNumber(&function->chunk, instruction);
269 
270  /* Print the simple stuff that we already know */
271  fprintf(stderr, " File \"%s\", line %d, in %s\n",
272  (function->chunk.filename ? function->chunk.filename->chars : "?"),
273  lineNo,
274  (function->name ? function->name->chars : "(unnamed)"));
275 
276 #ifndef NO_SOURCE_IN_TRACEBACK
277  /* Try to open the file */
278  if (function->chunk.filename) {
279  FILE * f = fopen(function->chunk.filename->chars, "r");
280  if (f) {
281  int line = 1;
282  do {
283  int c = fgetc(f);
284  if (c < -1) break;
285  if (c == '\n') {
286  line++;
287  continue;
288  }
289  if (line == lineNo) {
290  fprintf(stderr," ");
291  while (c == ' ') c = fgetc(f);
292  do {
293  fputc(c, stderr);
294  c = fgetc(f);
295  } while (!feof(f) && c > 0 && c != '\n');
296  fprintf(stderr, "\n");
297  break;
298  }
299  } while (!feof(f));
300  fclose(f);
301  }
302  }
303 #endif
304  }
305  }
306  }
307

309  if (krk_isInstanceOf(exception, vm.exceptions->syntaxError)) {
310  KrkValue result = krk_callDirect(krk_getType(exception)->_tostr, 1);
311  fprintf(stderr, "%s\n", AS_CSTRING(result));
312  return;
313  }

317  krk_currentThread.flags &= ~(KRK_THREAD_HAS_EXCEPTION);
318

324  fprintf(stderr, "%s.", AS_CSTRING(module));
325  }
326

328  fprintf(stderr, "%s", krk_typeName(exception));
329

332  if (!IS_STRING(result) || AS_STRING(result)->length == 0) {
333  fprintf(stderr, "\n");
334  } else {
335  fprintf(stderr, ": ");

338  }
339

341  krk_currentThread.flags |= hadException;

360 static void attachTraceback(void) {
361  if (IS_INSTANCE(krk_currentThread.currentException)) {
362  KrkInstance * theException = AS_INSTANCE(krk_currentThread.currentException);
363  KrkValue tracebackList;
364  if (krk_tableGet(&theException->fields, OBJECT_VAL(S("traceback")), &tracebackList)) {
365  krk_push(tracebackList);
366  } else {
367  krk_push(NONE_VAL());
368  }
369  tracebackList = krk_list_of(0,NULL,0);
370  krk_push(tracebackList);
371 
372  /* Build the traceback object */
373  if (krk_currentThread.frameCount) {
374 
375  /* Go up until we get to the exit frame */
376  size_t frameOffset = 0;
377  if (krk_currentThread.stackTop > krk_currentThread.stack) {
378  size_t stackOffset = krk_currentThread.stackTop - krk_currentThread.stack - 1;
379  while (stackOffset > 0 && !IS_HANDLER_TYPE(krk_currentThread.stack[stackOffset], OP_PUSH_TRY)) stackOffset--;
380  frameOffset = krk_currentThread.frameCount - 1;
381  while (frameOffset > 0 && krk_currentThread.frames[frameOffset].slots > stackOffset) frameOffset--;
382  }
383 
384  for (size_t i = frameOffset; i < krk_currentThread.frameCount; i++) {
385  KrkCallFrame * frame = &krk_currentThread.frames[i];
386  KrkTuple * tbEntry = krk_newTuple(2);
387  krk_push(OBJECT_VAL(tbEntry));
388  tbEntry->values.values[tbEntry->values.count++] = OBJECT_VAL(frame->closure);
389  tbEntry->values.values[tbEntry->values.count++] = INTEGER_VAL(frame->ip - frame->closure->function->chunk.code - 1);
390  krk_writeValueArray(AS_LIST(tracebackList), OBJECT_VAL(tbEntry));
391  krk_pop();
392  }
393  }
394 
395  if (IS_list(krk_peek(1))) {
396  KrkValueArray * existingTraceback = AS_LIST(krk_peek(1));
397  for (size_t i = 0; i < existingTraceback->count; ++i) {
398  krk_writeValueArray(AS_LIST(tracebackList), existingTraceback->values[i]);
399  }
400  }
401 
402  krk_attachNamedValue(&theException->fields, "traceback", tracebackList);
403  krk_pop();
404  krk_pop();
405  } /* else: probably a legacy 'raise str', just don't bother. */
406 }
