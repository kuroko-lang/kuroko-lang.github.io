91 static KrkToken makeIndentation() {
92  char reject = (peek() == ' ') ? '\t' : ' ';
93  while (!isAtEnd() && (peek() == ' ' || peek() == '\t')) advance();
94  if (isAtEnd()) return makeToken(TOKEN_EOF);
95  for (const char * start = scanner.start; start < scanner.cur; start++) {
96  if (*start == reject) return errorToken("Invalid mix of indentation.");
97  }
98  KrkToken out = makeToken(TOKEN_INDENTATION);
99  if (reject == ' ') out.length *= 8;
100  if (peek() == '#' || peek() == '\n') {
101  while (!isAtEnd() && peek() != '\n') advance();
102  scanner.startOfLine = 1;
103  return makeToken(TOKEN_RETRY);
104  }
105  return out;
106 }

108 static KrkToken string(char quoteMark) {
109  if (peek() == quoteMark && peekNext(1) == quoteMark) {
110  advance(); advance();
111  /* Big string */
112  while (!isAtEnd()) {
113  if (peek() == quoteMark && peekNext(1) == quoteMark && peekNext(2) == quoteMark) {
114  advance();
115  advance();
116  advance();
117  return makeToken(TOKEN_BIG_STRING);
118  }
119 
120  if (peek() == '\\') advance();
121  if (peek() == '\n') {
122  advance();
123  nextLine();
124  }
125  else advance();
126  }
127  if (isAtEnd()) return errorToken("Unterminated string.");
128  }
129  while (peek() != quoteMark && !isAtEnd()) {
130  if (peek() == '\n') return errorToken("Unterminated string.");
131  if (peek() == '\\') advance();
132  if (peek() == '\n') {
133  advance();
134  nextLine();

137  }

151 static KrkToken number(char c) {
152  if (c == '0') {
153  if (peek() == 'x' || peek() == 'X') {
154  /* Hexadecimal */
155  advance();
156  while (isDigit(peek()) || (peek() >= 'a' && peek() <= 'f') ||
157  (peek() >= 'A' && peek() <= 'F')) advance();
158  return makeToken(TOKEN_NUMBER);
159  } else if (peek() == 'b' || peek() == 'B') {
160  /* Binary */
161  advance();
162  while (peek() == '0' || peek() == '1') advance();
163  return makeToken(TOKEN_NUMBER);
164  } if (peek() == 'o' || peek() == 'O') {
165  /* Octal - must be 0o, none of those silly 0123 things */
166  advance();
167  while (peek() >= '0' && peek() <= '7') advance();
168  return makeToken(TOKEN_NUMBER);
169  }
170  /* Otherwise, decimal and maybe 0.123 floating */
171  }
172 
173  /* Decimal */
174  while (isDigit(peek())) advance();
175 
176  /* Floating point */
177  if (peek() == '.' && isDigit(peekNext(1))) {
178  advance();
179  while (isDigit(peek())) advance();
180  }
181 
182  return makeToken(TOKEN_NUMBER);
183 }

294  if (scanner.hasUnget) {
295  scanner.hasUnget = 0;
296  return scanner.unget;
297  }
298 
299  /* If at start of line, do thing */
300  if (scanner.startOfLine && (peek() == ' ' || peek() == '\t')) {
301  scanner.start = scanner.cur;
302  scanner.startOfLine = 0;
303  return makeIndentation();
304  }
305 
306  /* Eat whitespace */
307  skipWhitespace();
308 
309  /* Skip comments */
310  if (peek() == '#') while (peek() != '\n' && !isAtEnd()) advance();
311 
312  scanner.start = scanner.cur;
313  if (isAtEnd()) return makeToken(TOKEN_EOF);
314 
315  char c = advance();
316 
317  if (c == '\n') {
318  KrkToken out;
319  if (scanner.startOfLine) {
320  /* Ignore completely blank lines */
321  out = makeToken(TOKEN_RETRY);
322  } else {
323  scanner.startOfLine = 1;
324  out = makeToken(TOKEN_EOL);
325  }
326  nextLine();
327  return out;
328  }
329 
330  if (c == '\\' && peek() == '\n') {
331  advance();
332  nextLine();
333  return makeToken(TOKEN_RETRY);
334  }
335 
336  /* Not indentation, not a linefeed on an empty line, must be not be start of line any more */
337  scanner.startOfLine = 0;
338 
339  if (isAlpha(c) || (unsigned char)c > 0x7F) return identifier();
340  if (isDigit(c)) return number(c);
341 
342  switch (c) {
343  case '(': return makeToken(TOKEN_LEFT_PAREN);
344  case ')': return makeToken(TOKEN_RIGHT_PAREN);
345  case '{': return makeToken(TOKEN_LEFT_BRACE);
346  case '}': return makeToken(TOKEN_RIGHT_BRACE);
347  case '[': return makeToken(TOKEN_LEFT_SQUARE);
348  case ']': return makeToken(TOKEN_RIGHT_SQUARE);
349  case ',': return makeToken(TOKEN_COMMA);
350  case '.': return makeToken(TOKEN_DOT);
351  case ';': return makeToken(TOKEN_SEMICOLON);
352  case '@': return makeToken(TOKEN_AT);
353  case '~': return makeToken(TOKEN_TILDE);
354 
355  case ':': return makeToken(match('=') ? TOKEN_WALRUS : TOKEN_COLON);
356  case '!': return makeToken(match('=') ? TOKEN_BANG_EQUAL : TOKEN_BANG);
357  case '=': return makeToken(match('=') ? TOKEN_EQUAL_EQUAL : TOKEN_EQUAL);
358  case '<': return makeToken(match('=') ? TOKEN_LESS_EQUAL : (match('<') ? (match('=') ? TOKEN_LSHIFT_EQUAL : TOKEN_LEFT_SHIFT) : TOKEN_LESS));
359  case '>': return makeToken(match('=') ? TOKEN_GREATER_EQUAL : (match('>') ? (match('=') ? TOKEN_RSHIFT_EQUAL : TOKEN_RIGHT_SHIFT) : TOKEN_GREATER));
360  case '-': return makeToken(match('=') ? TOKEN_MINUS_EQUAL : (match('-') ? TOKEN_MINUS_MINUS : (match('>') ? TOKEN_ARROW : TOKEN_MINUS)));
361  case '+': return makeToken(match('=') ? TOKEN_PLUS_EQUAL : (match('+') ? TOKEN_PLUS_PLUS : TOKEN_PLUS));
362  case '^': return makeToken(match('=') ? TOKEN_CARET_EQUAL : TOKEN_CARET);
363  case '|': return makeToken(match('=') ? TOKEN_PIPE_EQUAL : TOKEN_PIPE);
364  case '&': return makeToken(match('=') ? TOKEN_AMP_EQUAL : TOKEN_AMPERSAND);
365  case '/': return makeToken(match('=') ? TOKEN_SOLIDUS_EQUAL : (match('/') ? (match('=') ? TOKEN_DSOLIDUS_EQUAL : TOKEN_DOUBLE_SOLIDUS) : TOKEN_SOLIDUS));
366  case '*': return makeToken(match('=') ? TOKEN_ASTERISK_EQUAL: (match('*') ? (match('=') ? TOKEN_POW_EQUAL : TOKEN_POW) : TOKEN_ASTERISK));
367  case '%': return makeToken(match('=') ? TOKEN_MODULO_EQUAL : TOKEN_MODULO);
