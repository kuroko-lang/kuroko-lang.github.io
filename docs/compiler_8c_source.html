866  } else {

1238 static void argumentDefinition(void) {
1239  if (match(TOKEN_EQUAL)) {
1240  /*
1241  * We inline default arguments by checking if they are equal
1242  * to a sentinel value and replacing them with the requested
1243  * argument. This allows us to send None (useful) to override
1244  * defaults that are something else. This essentially ends
1245  * up as the following at the top of the function:
1246  * if param == KWARGS_SENTINEL:
1247  * param = EXPRESSION
1248  */

1250  EMIT_OPERAND_OP(OP_GET_LOCAL, myLocal);
1251  emitBytes(OP_UNSET, OP_IS);
1252  int jumpIndex = emitJump(OP_JUMP_IF_FALSE_OR_POP);
1253  beginScope();
1254  expression(); /* Read expression */
1255  EMIT_OPERAND_OP(OP_SET_LOCAL, myLocal);
1256  endScope();
1257  patchJump(jumpIndex);
1258  emitByte(OP_POP); /* comparison result or expression value */
1259  current->codeobject->keywordArgs++;
1260  } else {
1261  if (current->codeobject->keywordArgs) {
1262  error("non-keyword argument follows keyword argument");
1263  return;
1264  }
1265  current->codeobject->requiredArgs++;
1266  }

1271  initCompiler(&compiler, type);
1272  compiler.codeobject->chunk.filename = compiler.enclosing->codeobject->chunk.filename;
1273 
1274  beginScope();
1275 
1276  if (isMethod(type)) current->codeobject->requiredArgs = 1;
1277  if (isCoroutine(type)) current->codeobject->flags |= KRK_CODEOBJECT_FLAGS_IS_COROUTINE;
1278 
1279  int hasCollectors = 0;
1280  KrkToken self = syntheticToken("self");
1281 
1282  consume(TOKEN_LEFT_PAREN, "Expected start of parameter list after function name.");
1283  startEatingWhitespace();
1284  if (!check(TOKEN_RIGHT_PAREN)) {
1285  do {
1286  if (isMethod(type) && check(TOKEN_IDENTIFIER) &&
1287  identifiersEqual(&parser.current, &self)) {
1288  if (hasCollectors || current->codeobject->requiredArgs != 1) {
1289  errorAtCurrent("Argument name 'self' in a method signature is reserved for the implicit first argument.");
1290  goto _bail;
1291  }
1292  advance();
1293  if (check(TOKEN_COLON)) {
1294  KrkToken name = parser.previous;
1295  match(TOKEN_COLON);
1296  typeHint(name);
1297  }
1298  if (check(TOKEN_EQUAL)) {
1299  errorAtCurrent("'self' can not be a keyword argument.");
1300  goto _bail;
1301  }
1302  continue;
1303  }
1304  if (match(TOKEN_ASTERISK) || check(TOKEN_POW)) {
1305  if (match(TOKEN_POW)) {
1306  if (hasCollectors == 2) {
1307  error("Duplicate ** in parameter list.");
1308  goto _bail;
1309  }
1310  hasCollectors = 2;
1311  current->codeobject->flags |= KRK_CODEOBJECT_FLAGS_COLLECTS_KWS;
1312  } else {
1313  if (hasCollectors) {
1314  error("Syntax error.");
1315  goto _bail;
1316  }
1317  hasCollectors = 1;
1318  current->codeobject->flags |= KRK_CODEOBJECT_FLAGS_COLLECTS_ARGS;
1319  }

1321  ssize_t paramConstant = parseVariable(
1322  (hasCollectors == 1) ? "Expected parameter name after '*'." : "Expected parameter name after '**'.");
1323  if (parser.hadError) goto _bail;
1324  defineVariable(paramConstant);
1325  if (check(TOKEN_COLON)) {
1326  KrkToken name = parser.previous;
1327  match(TOKEN_COLON);
1328  typeHint(name);
1329  }
1330  /* Make that a valid local for this function */
1331  size_t myLocal = current->localCount - 1;
1332  EMIT_OPERAND_OP(OP_GET_LOCAL, myLocal);
1333  /* Check if it's equal to the unset-kwarg-sentinel value */
1334  emitBytes(OP_UNSET, OP_IS);
1335  int jumpIndex = emitJump(OP_JUMP_IF_FALSE_OR_POP);
1336  /* And if it is, set it to the appropriate type */
1337  beginScope();
1338  if (hasCollectors == 1) EMIT_OPERAND_OP(OP_MAKE_LIST,0);
1339  else EMIT_OPERAND_OP(OP_MAKE_DICT,0);
1340  EMIT_OPERAND_OP(OP_SET_LOCAL, myLocal);
1341  endScope();

1343  patchJump(jumpIndex);
1344  emitByte(OP_POP); /* comparison value or expression */
1345  continue;
1346  }
1347  if (hasCollectors) {
1348  error("arguments follow catch-all collector");
1349  break;
1350  }
1351  ssize_t paramConstant = parseVariable("Expected parameter name.");
1352  if (parser.hadError) goto _bail;
1353  defineVariable(paramConstant);
1354  if (check(TOKEN_COLON)) {
1355  KrkToken name = parser.previous;
1356  match(TOKEN_COLON);
1357  typeHint(name);
1358  }
1359  argumentDefinition();
1360  } while (match(TOKEN_COMMA));
1361  }
1362  stopEatingWhitespace();
1363  consume(TOKEN_RIGHT_PAREN, "Expected end of parameter list.");
1364 
1365  if (match(TOKEN_ARROW)) {
