820 static void compareChained(struct GlobalState * state, int inner) {
821  KrkTokenType operatorType = state->parser.previous.type;
822  if (operatorType == TOKEN_NOT) consume(TOKEN_IN, "'in' must follow infix 'not'");
823  int invert = (operatorType == TOKEN_IS && match(TOKEN_NOT));
824 
825  ParseRule * rule = getRule(operatorType);
826  parsePrecedence(state, (Precedence)(rule->precedence + 1));
827 
828  if (getRule(state->parser.current.type)->precedence == PREC_COMPARISON) {
829  emitByte(OP_SWAP);
830  emitBytes(OP_DUP, 1);
831  }
832 
833  switch (operatorType) {
834  case TOKEN_BANG_EQUAL: emitBytes(OP_EQUAL, OP_NOT); break;
835  case TOKEN_EQUAL_EQUAL: emitByte(OP_EQUAL); break;
836  case TOKEN_GREATER: emitByte(OP_GREATER); break;
837  case TOKEN_GREATER_EQUAL: emitByte(OP_GREATER_EQUAL); break;
838  case TOKEN_LESS: emitByte(OP_LESS); break;
839  case TOKEN_LESS_EQUAL: emitByte(OP_LESS_EQUAL); break;
840 
841  case TOKEN_IS: emitByte(OP_IS); if (invert) emitByte(OP_NOT); break;
842 
843  case TOKEN_IN: emitByte(OP_INVOKE_CONTAINS); break;
844  case TOKEN_NOT: emitBytes(OP_INVOKE_CONTAINS, OP_NOT); break;
845 
846  default: error("Invalid binary comparison operator?"); break;
847  }
848 
849  if (getRule(state->parser.current.type)->precedence == PREC_COMPARISON) {
850  size_t exitJump = emitJump(OP_JUMP_IF_FALSE_OR_POP);
851  advance();
852  compareChained(state, 1);
853  patchJump(exitJump);
854  if (getRule(state->parser.current.type)->precedence != PREC_COMPARISON) {
855  if (!inner) {
856  emitBytes(OP_SWAP,OP_POP);
857  }
858  }
859  } else if (inner) {
860  emitByte(OP_JUMP);
861  emitBytes(0,2);
862  }
863 }

1377 static void argumentDefinition(struct GlobalState * state, int hasCollectors) {
1378  if (match(TOKEN_EQUAL)) {
1379  /*
1380  * We inline default arguments by checking if they are equal
1381  * to a sentinel value and replacing them with the requested
1382  * argument. This allows us to send None (useful) to override
1383  * defaults that are something else. This essentially ends
1384  * up as the following at the top of the function:
1385  * if param == KWARGS_SENTINEL:
1386  * param = EXPRESSION
1387  */
1388  size_t myLocal = state->current->localCount - 1;
1389  EMIT_OPERAND_OP(OP_GET_LOCAL, myLocal);
1390  int jumpIndex = emitJump(OP_TEST_ARG);
1391  beginScope(state);
1392  expression(state); /* Read expression */
1393  EMIT_OPERAND_OP(OP_SET_LOCAL_POP, myLocal);
1394  endScope(state);
1395  patchJump(jumpIndex);
1396  if (hasCollectors) {
1397  state->current->codeobject->keywordArgs++;
1398  } else {
1399  state->current->codeobject->potentialPositionals++;
1400  }
1401  } else {
1402  if (hasCollectors) {
1403  size_t myLocal = state->current->localCount - 1;
1404  EMIT_OPERAND_OP(OP_GET_LOCAL, myLocal);
1405  int jumpIndex = emitJump(OP_TEST_ARG);
1406  EMIT_OPERAND_OP(OP_MISSING_KW, state->current->codeobject->keywordArgs);
1407  patchJump(jumpIndex);
1408  state->current->codeobject->keywordArgs++;
1409  } else {
1410  if (state->current->codeobject->potentialPositionals != state->current->codeobject->requiredArgs) {
1411  error("non-default argument follows default argument");
1412  return;
1413  }
1414  state->current->codeobject->requiredArgs++;
1415  state->current->codeobject->potentialPositionals++;
1416  }
1417  }
1418 }

1494  }

1496  match(TOKEN_COLON);
1497  typeHint(state, name);

1503  int jumpIndex = emitJump(OP_TEST_ARG);

1506  if (hasCollectors == 1) EMIT_OPERAND_OP(OP_MAKE_LIST,0);
1507  else EMIT_OPERAND_OP(OP_MAKE_DICT,0);
1508  EMIT_OPERAND_OP(OP_SET_LOCAL_POP, myLocal);
1509  endScope(state);

1511  patchJump(jumpIndex);
1512  continue;
1513  }
1514  if (hasCollectors == 2) {
1515  error("arguments follow catch-all keyword collector");
1516  break;
1517  }
1518  ssize_t paramConstant = parseVariable(state, "Expected parameter name.");
1519  if (state->parser.hadError) return 1;
1520  hideLocal(state);
1521  if (type != TYPE_LAMBDA && check(TOKEN_COLON)) {
1522  KrkToken name = state->parser.previous;
1523  match(TOKEN_COLON);
1524  typeHint(state, name);
1525  }

3830 static void parsePrecedence(struct GlobalState * state, Precedence precedence) {
3831  RewindState rewind = {recordChunk(currentChunk()), krk_tellScanner(&state->scanner), state->parser};
3832 
3833  advance();
3834  ParseFn prefixRule = getRule(state->parser.previous.type)->prefix;
3835 
3836  /* Only allow *expr parsing where we would allow comma expressions,
3837  * otherwise pretend this prefix rule doesn't exist. */
3838  if (prefixRule == pstar && precedence > PREC_COMMA) prefixRule = NULL;
3839 
3840  if (prefixRule == NULL) {
3841  switch (state->parser.previous.type) {
3842  case TOKEN_RIGHT_BRACE:
3843  case TOKEN_RIGHT_PAREN:
3844  case TOKEN_RIGHT_SQUARE:
3845  error("Unmatched '%.*s'",
3846  (int)state->parser.previous.length, state->parser.previous.start);
3847  break;
3848  case TOKEN_EOL:
3849  /* TODO: This should definitely be tripping the REPL to ask for more input. */
3850  error("Unexpected end of line");
3851  break;
3852  case TOKEN_EOF:
3853  error("Unexpected end of input");
3854  break;
3855  default:
3856  error("'%.*s' does not start an expression",
3857  (int)state->parser.previous.length, state->parser.previous.start);
3858  }
3859  return;
3860  }
3861  int exprType = 0;
3862  if (precedence <= PREC_ASSIGNMENT || precedence == PREC_CAN_ASSIGN) exprType = EXPR_CAN_ASSIGN;
3863  if (precedence == PREC_MUST_ASSIGN) exprType = EXPR_ASSIGN_TARGET;
3864  if (precedence == PREC_DEL_TARGET) exprType = EXPR_DEL_TARGET;
3865  prefixRule(state, exprType, &rewind);
3866  while (precedence <= getRule(state->parser.current.type)->precedence) {
3867  if (state->parser.hadError) {
3868  skipToEnd();
3869  return;
3870  }
3871 
3872  if (exprType == EXPR_ASSIGN_TARGET && (state->parser.previous.type == TOKEN_COMMA ||
3873  state->parser.previous.type == TOKEN_EQUAL)) break;
3874  advance();
3875  ParseFn infixRule = getRule(state->parser.previous.type)->infix;
3876  infixRule(state, exprType, &rewind);
3877  }
3878 
3879  if (exprType == EXPR_CAN_ASSIGN && matchAssignment(state)) {
3880  error("Invalid assignment target");
3881  }

3885 static int maybeSingleExpression(struct GlobalState * state) {
3886  /* We're only going to use this to reset if we found a string, and it turns out
3887  * not to be a docstring. */
3888  RewindState rewind = {recordChunk(currentChunk()), krk_tellScanner(&state->scanner), state->parser};
3889 
3897  if (check(TOKEN_STRING) || check(TOKEN_BIG_STRING)) {
3898  advance();
3899  if (match(TOKEN_EOL)) {
3900  /* We found just a string on the first line, but is it the only line?
3901  * We should treat that as a regular string expression, eg. in a repl. */
3902  int isEof = check(TOKEN_EOF);
3903  /* Regardless, restore the scanner/parser so we can actually parse the string. */
3904  krk_rewindScanner(&state->scanner, rewind.oldScanner);
3905  state->parser = rewind.oldParser;
3906  advance();
3907  /* Parse the string. */
3908  string(state, EXPR_NORMAL, NULL);
3909  /* If we did see end of input, it's a simple string expression. */
3910  if (isEof) return 1;
3911  /* Otherwise, it's a docstring, and there's more code following it.
3912  * Emit the instructions to assign the docstring to the current globals. */
3913  KrkToken doc = syntheticToken("__doc__");
3914  size_t ind = identifierConstant(state, &doc);
3915  EMIT_OPERAND_OP(OP_DEFINE_GLOBAL, ind);
3916  return 0;
3917  } else {
3918  /* There was something other than a line feed after the string token,
3919  * rewind so we can parse as an expression next. */
3920  krk_rewindScanner(&state->scanner, rewind.oldScanner);
3921  state->parser = rewind.oldParser;
3922  }
3923  }
3924 
3925  /* Try to parse one single expression */
3926  ParseRule * rule = getRule(state->parser.current.type);
3927  if (rule->prefix) {
3928  parsePrecedence(state, PREC_ASSIGNMENT);
3929 
3930  /* Semicolon after expression statement, finish this one and continue
3931  * parsing only more simple statements, as we would normally. */
3932  if (match(TOKEN_SEMICOLON)) {
3933  emitByte(OP_POP);
3934  simpleStatement(state);
3935  return 0;
3936  }
3937 
3938  /* Expression statement that isn't the end of the input, finish it
3939  * and let the declaration loop handle the rest. */
3940  if (match(TOKEN_EOL) && !check(TOKEN_EOF)) {
3941  emitByte(OP_POP);
3942  return 0;
3943  }
3944 
3945  /* End of input after expression, must be just the single expression;
3946  * using check rather than match makes the return emit on the same
3947  * line, which produces cleaner disassembly when -d tracing is enabled. */
3948  if (check(TOKEN_EOF))
3949  return 1;
3950 
3951  /* Must be an error. */
3952  errorAfterStatement(state);
3953  return 0;
3954  }
3955 
3956  return 0;
3957 }
