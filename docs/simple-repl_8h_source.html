59  * set up to let us compile "on the fly" so we can't just
60  * run lines through it and see if it wants more... */
61  if (lineLength > 1 && lines[i][lineLength-2] == ':') {

63  blockWidth = countSpaces + 4;
64  continue;
65  } else if (lineLength > 1 && lines[i][lineLength-2] == '\\') {
66  inBlock = 1;
67  continue;
68  } else if (inBlock && lineLength != 1) {
69  if (isSpaces) {
70  free(lines[i]);
71  totalData -= lineLength;
72  lineCount--;
73  break;
74  }
75  blockWidth = countSpaces;
76  continue;
77  } else if (lineLength > 1 && lines[i][countSpaces] == '@') {
78  inBlock = 1;
79  blockWidth = countSpaces;
80  continue;
81  }
82 
83  /* Ignore blank lines. */
84  if (isSpaces && !i) valid = 0;
85 
86  /* If we're not in a block, or have entered a blank line,
87  * we can stop reading new lines and jump to execution. */
88  break;
89  }
90

105  const char * formatStr = " \033[1;30m=> %s\033[0m\n";
106  if (type->_reprer) {

109  } else if (type->_tostr) {

112  }
113  if (!IS_STRING(result)) {

115  } else {

117  }

123  }
