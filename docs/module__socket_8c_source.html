112 static int socket_parse_address(struct socket * self, KrkValue address, struct sockaddr_storage *sock_addr, socklen_t *sock_size) {
113  if (self->family == AF_INET) {
114  /* Should be 2-tuple */
115  if (!IS_tuple(address)) {
116  krk_runtimeError(vm.exceptions->typeError, "Expected 2-tuple, not '%s'", krk_typeName(address));
117  return 1;
118  }
119  KrkTuple * addr = AS_TUPLE(address);
120  if (addr->values.count != 2) {
121  krk_runtimeError(vm.exceptions->typeError, "Expected 2-tuple, not '%s'", krk_typeName(address));
122  return 1;
123  }
124  if (!IS_str(addr->values.values[0])) {
125  krk_runtimeError(vm.exceptions->typeError, "Address should be int, not '%s'", krk_typeName(addr->values.values[0]));
126  return 1;
127  }
128  if (!IS_int(addr->values.values[1])) {
129  krk_runtimeError(vm.exceptions->typeError, "Port should be int, not '%s'", krk_typeName(addr->values.values[1]));
130  return 1;
131  }
132 
133  if (!AS_STRING(addr->values.values[0])->length) {
134  struct sockaddr_in * sin = (struct sockaddr_in*)sock_addr;
135  *sock_size = sizeof(struct sockaddr_in);
136  sin->sin_family = AF_INET;
137  sin->sin_port = htons(AS_int(addr->values.values[1]));
138  sin->sin_addr.s_addr = INADDR_ANY;
139  return 0;
140  } else {
141  struct addrinfo *result;
142  struct addrinfo *res;
143  int error = getaddrinfo(AS_CSTRING(addr->values.values[0]), NULL, NULL, &result);
144  if (error != 0) {
145  krk_runtimeError(SocketError, "getaddrinfo() returned error: %d", error);
146  return 1;
147  }
148 
149  int found = 0;
150  res = result;
151  while (res) {
152  if (res->ai_family == AF_INET) {
153  found = 1;
154  *sock_size = res->ai_addrlen;
155  memcpy(sock_addr, res->ai_addr, *sock_size);
156  break;
157  }
158  res = res->ai_next;
159  }
160 
161  freeaddrinfo(result);
162 
163  if (!found) {
164  krk_runtimeError(SocketError, "no suitable address");
165  return 1;
166  }
167 
168  struct sockaddr_in * sin = (struct sockaddr_in*)sock_addr;
169  sin->sin_family = AF_INET;
170  sin->sin_port = htons(AS_int(addr->values.values[1]));
171 
172  return 0;
173  }
174  } else {
175  krk_runtimeError(vm.exceptions->notImplementedError, "Not implemented.");
176  return 1;
177  }
178 
179  return 1;
180 }
