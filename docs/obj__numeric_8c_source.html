187 static KrkValue _krk_int_div(krk_integer_type a, krk_integer_type b) {
188  if (unlikely(b == 0)) return krk_runtimeError(vm.exceptions->zeroDivisionError, "integer division or modulo by zero");
189  if (a == 0) return INTEGER_VAL(0);
190  int64_t abs_a = a < 0 ? -a : a;
191  int64_t abs_b = b < 0 ? -b : b;
192  if ((a < 0) != (b < 0)) {
193  /* If signs don't match, the result is negative, and rounding down means away from 0... */
194  int64_t res = -1 - (abs_a - 1) / abs_b;
195  return INTEGER_VAL(res);
196  }
197  return INTEGER_VAL((abs_a / abs_b));
198 }

200 static KrkValue _krk_int_mod(krk_integer_type a, krk_integer_type b) {
201  if (unlikely(b == 0)) return krk_runtimeError(vm.exceptions->zeroDivisionError, "integer division or modulo by zero");
202  if (a == 0) return INTEGER_VAL(0);
203  int64_t abs_a = a < 0 ? -a : a;
204  int64_t abs_b = b < 0 ? -b : b;
205  int64_t res;
206  if ((a < 0) != (b < 0)) {
207  /* If quotient would be negative, then remainder is inverted against the divisor. */
208  res = (abs_b - 1 - (abs_a - 1) % abs_b);
209  } else {
210  res = abs_a % abs_b;
211  }
212  /* Negative divisor always yields negative remainder, except when it's 0... */
213  return INTEGER_VAL((b < 0) ? -res : res);
214 
215 }

271  struct StringBuilder sb = {0};

276  val = val >> 1;
