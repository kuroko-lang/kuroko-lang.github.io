58 KRK_Method(bytes,__hash__) {
59  METHOD_TAKES_NONE();
60  uint32_t hash = 0;
61  /* This is the so-called "sdbm" hash. It comes from a piece of
62  * public domain code from a clone of ndbm. */
63  for (size_t i = 0; i < self->length; ++i) {
64  hash = (int)self->bytes[i] + (hash << 6) + (hash << 16) - hash;
65  }
66  return INTEGER_VAL(hash);
67 }
