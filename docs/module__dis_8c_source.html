40  int lineNo;

42  if (!krk_parseArgs("Vi|i",(const char*[]){"func","lineno","flags"}, &func, &lineNo, &flags)) return NONE_VAL();

50  }
51  }
52 
53  int result;
54  if (IS_STRING(func)) {
55  result = krk_debug_addBreakpointFileLine(AS_STRING(func), lineNo, flags);
56  } else {
57  KrkCodeObject * target = NULL;
58  if (IS_CLOSURE(func)) {
59  target = AS_CLOSURE(func)->function;
60  } else if (IS_BOUND_METHOD(func) && IS_CLOSURE(OBJECT_VAL(AS_BOUND_METHOD(func)->method))) {
61  target = AS_CLOSURE(OBJECT_VAL(AS_BOUND_METHOD(func)->method))->function;
62  } else if (IS_codeobject(func)) {
63  target = AS_codeobject(func);
64  } else {

66  }
67  /* Figure out what instruction this should be on */
68  size_t last = 0;
69  for (size_t i = 0; i < target->chunk.linesCount; ++i) {
70  if (target->chunk.lines[i].line > (size_t)lineNo) break;
71  if (target->chunk.lines[i].line == (size_t)lineNo) {
72  last = target->chunk.lines[i].startOffset;
73  break;
74  }
75  last = target->chunk.lines[i].startOffset;
76  }
77  result = krk_debug_addBreakpointCodeOffset(target,last,flags);
78  }
79 
80  if (result < 0)
81  return krk_runtimeError(vm.exceptions->baseException, "Could not add breakpoint.");
82 
83  return INTEGER_VAL(result);

166 static KrkValue _examineInternal(KrkCodeObject* func) {
167  KrkValue output = krk_list_of(0,NULL,0);
168  krk_push(output);
169 
170  KrkChunk * chunk = &func->chunk;
171  size_t offset = 0;
172  while (offset < chunk->count) {
173  uint8_t opcode = chunk->code[offset];
174  size_t size = 0;
175  ssize_t constant = -1;
176  ssize_t jump = 0;
177  ssize_t operand = -1;
178  ssize_t local = -1;
179  switch (opcode) {
180 #include "opcodes.h"
181  }
182 
183  KrkTuple * newTuple = krk_newTuple(3);
184  krk_push(OBJECT_VAL(newTuple));
185  newTuple->values.values[newTuple->values.count++] = INTEGER_VAL(opcode);
186  newTuple->values.values[newTuple->values.count++] = INTEGER_VAL(size);
187  if (constant != -1) {
188  newTuple->values.values[newTuple->values.count++] = chunk->constants.values[constant];
189  } else if (jump != 0) {
190  newTuple->values.values[newTuple->values.count++] = INTEGER_VAL(jump);
191  } else if (local != -1) {
192  newTuple->values.values[newTuple->values.count++] = INTEGER_VAL(operand); /* Just in case */
193  for (size_t i = 0; i < func->localNameCount; ++i) {
194  if (func->localNames[i].id == (size_t)local && func->localNames[i].birthday <= offset && func->localNames[i].deathday >= offset) {
195  newTuple->values.values[newTuple->values.count-1] = OBJECT_VAL(func->localNames[i].name);
196  break;
197  }
198  }
199  } else if (operand != -1) {
200  newTuple->values.values[newTuple->values.count++] = INTEGER_VAL(operand);
201  } else {
202  newTuple->values.values[newTuple->values.count++] = NONE_VAL();
203  }
204  krk_writeValueArray(AS_LIST(output), krk_peek(0));
205  krk_pop();
206 
207  if (size == 0) {
208  abort();
209  }
210 
211  offset += size;
212  }
213 
214  return krk_pop();
215 }
