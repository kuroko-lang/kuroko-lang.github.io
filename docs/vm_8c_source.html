145 void krk_dumpTraceback() {
146  if (!krk_valuesEqual(krk_currentThread.currentException,NONE_VAL())) {
147  krk_push(krk_currentThread.currentException);
148  KrkValue tracebackEntries;
149  if (IS_INSTANCE(krk_currentThread.currentException)
150  && krk_tableGet(&AS_INSTANCE(krk_currentThread.currentException)->fields, OBJECT_VAL(S("traceback")), &tracebackEntries)
151  && IS_list(tracebackEntries) && AS_LIST(tracebackEntries)->count > 0) {
152  /* This exception has a traceback we can print. */
153  fprintf(stderr, "Traceback (most recent call last):\n");
154  for (size_t i = 0; i < AS_LIST(tracebackEntries)->count; ++i) {
155 
156  /* Quietly skip invalid entries as we don't want to bother printing explanatory text for them */
157  if (!IS_TUPLE(AS_LIST(tracebackEntries)->values[i])) continue;
158  KrkTuple * entry = AS_TUPLE(AS_LIST(tracebackEntries)->values[i]);
159  if (entry->values.count != 2) continue;
160  if (!IS_CLOSURE(entry->values.values[0])) continue;
161  if (!IS_INTEGER(entry->values.values[1])) continue;
162 
163  /* Get the function and instruction index from this traceback entry */
164  KrkClosure * closure = AS_CLOSURE(entry->values.values[0]);
165  KrkCodeObject * function = closure->function;
166  size_t instruction = AS_INTEGER(entry->values.values[1]);
167 
168  /* Calculate the line number */
169  int lineNo = (int)krk_lineNumber(&function->chunk, instruction);
170 
171  /* Print the simple stuff that we already know */
172  fprintf(stderr, " File \"%s\", line %d, in %s\n",
173  (function->chunk.filename ? function->chunk.filename->chars : "?"),
174  lineNo,
175  (function->name ? function->name->chars : "(unnamed)"));
176 
177  /* Try to open the file */
178  if (function->chunk.filename) {
179  FILE * f = fopen(function->chunk.filename->chars, "r");
180  if (f) {
181  int line = 1;
182  do {
183  char c = fgetc(f);
184  if (c < -1) break;
185  if (c == '\n') {
186  line++;
187  continue;
188  }
189  if (line == lineNo) {
190  fprintf(stderr," ");
191  while (c == ' ') c = fgetc(f);
192  do {
193  fputc(c, stderr);
194  c = fgetc(f);
195  } while (!feof(f) && c > 0 && c != '\n');
196  fprintf(stderr, "\n");
197  break;
198  }
199  } while (!feof(f));
200  fclose(f);
201  }
202  }
203  }
204  }
205 
206  /* Is this a SyntaxError? Handle those specially. */
207  if (krk_isInstanceOf(krk_currentThread.currentException, vm.exceptions->syntaxError)) {
208  KrkValue result = krk_callSimple(OBJECT_VAL(krk_getType(krk_currentThread.currentException)->_tostr), 1, 0);
209  fprintf(stderr, "%s\n", AS_CSTRING(result));
210  return;
211  }
212  /* Clear the exception state while printing the exception. */
213  krk_currentThread.flags &= ~(KRK_THREAD_HAS_EXCEPTION);
214  fprintf(stderr, "%s", krk_typeName(krk_currentThread.currentException));
215  KrkValue result = krk_callSimple(OBJECT_VAL(krk_getType(krk_currentThread.currentException)->_tostr), 1, 0);
216  if (!IS_STRING(result)) {
217  fprintf(stderr, "\n");
218  } else {
219  fprintf(stderr, ": %s\n", AS_CSTRING(result));
220  }
221  /* Turn the exception flag back on */
222  krk_currentThread.flags |= KRK_THREAD_HAS_EXCEPTION;
223  }

229 static void attachTraceback(void) {
230  if (IS_INSTANCE(krk_currentThread.currentException)) {
231  KrkInstance * theException = AS_INSTANCE(krk_currentThread.currentException);
232 
233  /* If there already is a traceback, don't add a new one; this exception was re-raised. */
234  KrkValue existing;
235  if (krk_tableGet(&theException->fields, OBJECT_VAL(S("traceback")), &existing)) return;
236 
237  KrkValue tracebackList = krk_list_of(0,NULL,0);
238  krk_push(tracebackList);
239  /* Build the traceback object */
240  if (krk_currentThread.frameCount) {
241  for (size_t i = 0; i < krk_currentThread.frameCount; i++) {
242  KrkCallFrame * frame = &krk_currentThread.frames[i];
243  KrkTuple * tbEntry = krk_newTuple(2);
244  krk_push(OBJECT_VAL(tbEntry));
245  tbEntry->values.values[tbEntry->values.count++] = OBJECT_VAL(frame->closure);
246  tbEntry->values.values[tbEntry->values.count++] = INTEGER_VAL(frame->ip - frame->closure->function->chunk.code - 1);
247  krk_writeValueArray(AS_LIST(tracebackList), OBJECT_VAL(tbEntry));
248  krk_pop();
249  }
250  }
251  krk_attachNamedValue(&theException->fields, "traceback", tracebackList);
252  krk_pop();
253  } /* else: probably a legacy 'raise str', just don't bother. */
254 }

617  size_t offsetOfExtraArgs = closure->function->requiredArgs + closure->function->keywordArgs;
618  size_t argCountX = argCount;
619  KrkValueArray * positionals;
620  KrkTable * keywords;
621 
622  if (argCount && IS_KWARGS(krk_currentThread.stackTop[-1])) {
623 
624  KrkValue myList = krk_list_of(0,NULL,0);
625  krk_currentThread.scratchSpace[0] = myList;
626  KrkValue myDict = krk_dict_of(0,NULL,0);
627  krk_currentThread.scratchSpace[1] = myDict;
628  positionals = AS_LIST(myList);
629  keywords = AS_DICT(myDict);
630 
631  /* This processes the existing argument list into a ValueArray and a Table with the args and keywords */
632  if (!krk_processComplexArguments(argCount, positionals, keywords, closure->function->name ? closure->function->name->chars : "<unnamed>")) goto _errorDuringPositionals;
633  argCount--; /* It popped the KWARGS value from the top, so we have one less argument */
634 
635  /* Do we already know we have too many arguments? Let's bail before doing a bunch of work. */
636  if ((positionals->count > potentialPositionalArgs) && !(closure->function->flags & KRK_CODEOBJECT_FLAGS_COLLECTS_ARGS)) {
637  checkArgumentCount(closure,positionals->count);
638  goto _errorDuringPositionals;
639  }
640 
641  /* Prepare stack space for all potential positionals, mark them unset */
642  for (size_t i = 0; i < (size_t)argCount; ++i) {
643  krk_currentThread.stackTop[-argCount + i] = KWARGS_VAL(0);
644  }
645 
646  /* Do we have a bunch of unused keyword argument slots? Fill them in. */
647  while ((size_t)argCount < potentialPositionalArgs) {
648  krk_push(KWARGS_VAL(0));
649  argCount++;
650  }
651 
652  /* Did we have way more arguments than we needed? Put the stack where it should be. */
653  while ((size_t)argCount > potentialPositionalArgs) {
654  krk_pop();
655  argCount--;
656  }
657 
658  /* Place positional arguments */
659  for (size_t i = 0; i < potentialPositionalArgs && i < positionals->count; ++i) {
660  krk_currentThread.stackTop[-argCount + i] = positionals->values[i];
661  }
662 
663  if (closure->function->flags & KRK_CODEOBJECT_FLAGS_COLLECTS_ARGS) {
664  size_t count = (positionals->count > potentialPositionalArgs) ? (positionals->count - potentialPositionalArgs) : 0;
665  KrkValue * offset = (count == 0) ? NULL : &positionals->values[potentialPositionalArgs];
666  krk_push(krk_list_of(count, offset, 0));
667  argCount++;
668  }
669 
670  krk_freeValueArray(positionals);
671  krk_currentThread.scratchSpace[0] = NONE_VAL();
672 
673  /* Now place keyword arguments */
674  for (size_t i = 0; i < keywords->capacity; ++i) {
675  KrkTableEntry * entry = &keywords->entries[i];
676  if (!IS_KWARGS(entry->key)) {
677  KrkValue name = entry->key;
678  KrkValue value = entry->value;
679  /* See if we can place it */
680  for (int j = 0; j < (int)closure->function->requiredArgs; ++j) {
681  if (krk_valuesEqual(name, closure->function->requiredArgNames.values[j])) {
682  if (!IS_KWARGS(krk_currentThread.stackTop[-argCount + j])) {
683  multipleDefs(closure,j);
684  goto _errorAfterPositionals;
685  }
686  krk_currentThread.stackTop[-argCount + j] = value;
687  goto _finishKwarg;
688  }
689  }
690  /* See if it's a keyword arg. */
691  for (int j = 0; j < (int)closure->function->keywordArgs; ++j) {
692  if (krk_valuesEqual(name, closure->function->keywordArgNames.values[j])) {
693  if (!IS_KWARGS(krk_currentThread.stackTop[-argCount + j + closure->function->requiredArgs])) {
694  multipleDefs(closure, j + closure->function->requiredArgs);
695  goto _errorAfterPositionals;
696  }
697  krk_currentThread.stackTop[-argCount + j + closure->function->requiredArgs] = value;
698  goto _finishKwarg;
699  }
700  }
701  if (!(closure->function->flags & KRK_CODEOBJECT_FLAGS_COLLECTS_KWS)) {
702  krk_runtimeError(vm.exceptions->typeError, "%s() got an unexpected keyword argument '%s'",
703  closure->function->name ? closure->function->name->chars : "<unnamed>",
704  AS_CSTRING(name));
705  goto _errorAfterPositionals;
706  }
707  continue;
708 _finishKwarg:
709  entry->key = KWARGS_VAL(0);
710  entry->value = BOOLEAN_VAL(1);
711  continue;
712  }
713  }
714 
715  /* If this function takes a **kwargs, we need to provide it as a dict */
716  if (closure->function->flags & KRK_CODEOBJECT_FLAGS_COLLECTS_KWS) {
717  krk_push(krk_dict_of(0,NULL,0));
718  argCount++;
719  krk_tableAddAll(keywords, AS_DICT(krk_peek(0)));
720  }
721 
722  krk_freeTable(keywords);
723  krk_currentThread.scratchSpace[1] = NONE_VAL();
724 
725  for (size_t i = 0; i < (size_t)closure->function->requiredArgs; ++i) {
726  if (IS_KWARGS(krk_currentThread.stackTop[-argCount + i])) {
727  krk_runtimeError(vm.exceptions->typeError, "%s() missing required positional argument: '%s'",
728  closure->function->name ? closure->function->name->chars : "<unnamed>",
729  AS_CSTRING(closure->function->requiredArgNames.values[i]));
730  goto _errorAfterKeywords;
731  }
732  }
733 
734  argCountX = argCount - (!!(closure->function->flags & KRK_CODEOBJECT_FLAGS_COLLECTS_ARGS) + !!(closure->function->flags & KRK_CODEOBJECT_FLAGS_COLLECTS_KWS));
735  } else {
736  /* We can't have had any kwargs. */
737  if ((size_t)argCount > potentialPositionalArgs && (closure->function->flags & KRK_CODEOBJECT_FLAGS_COLLECTS_ARGS)) {
738  krk_push(NONE_VAL()); krk_push(NONE_VAL()); krk_pop(); krk_pop();
739  startOfPositionals = &krk_currentThread.stackTop[-argCount];
740  KrkValue tmp = krk_list_of(argCount - potentialPositionalArgs,
741  &startOfPositionals[potentialPositionalArgs], 0);
742  startOfPositionals = &krk_currentThread.stackTop[-argCount];
743  startOfPositionals[offsetOfExtraArgs] = tmp;
744  argCount = closure->function->requiredArgs + 1;
745  argCountX = argCount - 1;
746  while (krk_currentThread.stackTop > startOfPositionals + argCount) krk_pop();
747  }
748  }
749  if (unlikely(!checkArgumentCount(closure, argCountX))) {
750  return 0;
751  }
752  while (argCount < (int)totalArguments) {
753  krk_push(KWARGS_VAL(0));
754  argCount++;
755  }
756  if (unlikely(closure->function->flags & (KRK_CODEOBJECT_FLAGS_IS_GENERATOR | KRK_CODEOBJECT_FLAGS_IS_COROUTINE))) {
757  KrkInstance * gen = krk_buildGenerator(closure, krk_currentThread.stackTop - argCount, argCount);
758  krk_currentThread.stackTop = krk_currentThread.stackTop - argCount - extra;
759  krk_push(OBJECT_VAL(gen));
760  return 2;
761  }
762  if (unlikely(krk_currentThread.frameCount == KRK_CALL_FRAMES_MAX)) {
763  krk_runtimeError(vm.exceptions->baseException, "Too many call frames.");
764  return 0;
765  }
766  KrkCallFrame * frame = &krk_currentThread.frames[krk_currentThread.frameCount++];
767  frame->closure = closure;
768  frame->ip = closure->function->chunk.code;
769  frame->slots = (krk_currentThread.stackTop - argCount) - krk_currentThread.stack;
770  frame->outSlots = (krk_currentThread.stackTop - argCount - extra) - krk_currentThread.stack;
771  frame->globals = &closure->function->globalsContext->fields;
772  FRAME_IN(frame);
773  return 1;
774 
775 _errorDuringPositionals:
776  krk_freeValueArray(positionals);
777  krk_currentThread.scratchSpace[0] = NONE_VAL();
778 _errorAfterPositionals:
779  krk_freeTable(keywords);
780  krk_currentThread.scratchSpace[1] = NONE_VAL();
781 _errorAfterKeywords:
782  return 0;
783 }

804 int krk_callValue(KrkValue callee, int argCount, int extra) {
805  if (IS_OBJECT(callee)) {
806  switch (OBJECT_TYPE(callee)) {
807  case KRK_OBJ_CLOSURE:
808  return call(AS_CLOSURE(callee), argCount, extra);
809  case KRK_OBJ_NATIVE: {
810  NativeFn native = (NativeFn)AS_NATIVE(callee)->function;
811  if (argCount && IS_KWARGS(krk_currentThread.stackTop[-1])) {
812  KrkValue myList = krk_list_of(0,NULL,0);
813  krk_currentThread.scratchSpace[0] = myList;
814  KrkValue myDict = krk_dict_of(0,NULL,0);
815  krk_currentThread.scratchSpace[1] = myDict;
816  if (!krk_processComplexArguments(argCount, AS_LIST(myList), AS_DICT(myDict), AS_NATIVE(callee)->name)) {
817  return 0;
818  }
819  argCount--; /* Because that popped the kwargs value */
820  krk_currentThread.stackTop -= argCount + extra; /* We can just put the stack back to normal */
821  krk_push(myList);
822  krk_push(myDict);
823  krk_currentThread.scratchSpace[0] = NONE_VAL();
824  krk_currentThread.scratchSpace[1] = NONE_VAL();
825  krk_writeValueArray(AS_LIST(myList), myDict);
826  KrkValue result = native(AS_LIST(myList)->count-1, AS_LIST(myList)->values, 1);
827  if (krk_currentThread.stackTop == krk_currentThread.stack) return 0;
828  krk_pop();
829  krk_pop();
830  krk_push(result);
831  } else {
832  KrkValue result;
833  if (argCount < 9) {
834  KrkValue stackCopy[8];
835  memcpy(stackCopy, krk_currentThread.stackTop - argCount, argCount * sizeof(KrkValue));
836  result = native(argCount, stackCopy, 0);
837  } else {
838  KrkValue * stackCopy = malloc(argCount * sizeof(KrkValue));
839  memcpy(stackCopy, krk_currentThread.stackTop - argCount, argCount * sizeof(KrkValue));
840  result = native(argCount, stackCopy, 0);
841  free(stackCopy);
842  }
843  if (krk_currentThread.stackTop == krk_currentThread.stack) return 0;
844  krk_currentThread.stackTop -= argCount + extra;
845  krk_push(result);
846  }
847  return 2;
848  }
849  case KRK_OBJ_INSTANCE: {
850  KrkClass * _class = AS_INSTANCE(callee)->_class;
851  KrkValue callFunction;
852  if (_class->_call) {
853  return krk_callValue(OBJECT_VAL(_class->_call), argCount + 1, 0);
854  } else if (krk_tableGet(&_class->methods, vm.specialMethodNames[METHOD_CALL], &callFunction)) {
855  return krk_callValue(callFunction, argCount + 1, 0);
856  } else {
857  krk_runtimeError(vm.exceptions->typeError, "'%s' object is not callable", krk_typeName(callee));
858  return 0;
859  }
860  }
861  case KRK_OBJ_CLASS: {
862  KrkClass * _class = AS_CLASS(callee);
863  KrkInstance * newInstance = krk_newInstance(_class);
864  krk_currentThread.stackTop[-argCount - 1] = OBJECT_VAL(newInstance);
865  KrkValue initializer;
866  if (_class->_init) {
867  return krk_callValue(OBJECT_VAL(_class->_init), argCount + 1, 0);
868  } else if (krk_tableGet(&_class->methods, vm.specialMethodNames[METHOD_INIT], &initializer)) {
869  return krk_callValue(initializer, argCount + 1, 0);
870  } else if (argCount != 0) {
871  krk_runtimeError(vm.exceptions->typeError, "%s() takes no arguments (%d given)",
872  _class->name->chars, argCount);
873  return 0;
874  }
875  return 1;
876  }
877  case KRK_OBJ_BOUND_METHOD: {
878  KrkBoundMethod * bound = AS_BOUND_METHOD(callee);
879  krk_currentThread.stackTop[-argCount - 1] = bound->receiver;
880  if (!bound->method) {
881  krk_runtimeError(vm.exceptions->argumentError, "???");
882  return 0;
883  }
884  return krk_callValue(OBJECT_VAL(bound->method), argCount + 1, 0);
885  }
886  default:
887  break;
888  }
889  }
890  krk_runtimeError(vm.exceptions->typeError, "'%s' object is not callable", krk_typeName(callee));
891  return 0;
892 }

912 int krk_bindMethod(KrkClass * _class, KrkString * name) {
913  KrkValue method, out;
914  while (_class) {
915  if (krk_tableGet_fast(&_class->methods, name, &method)) break;
916  _class = _class->base;
917  }
918  if (!_class) return 0;
919  if (IS_NATIVE(method)) {
920  if (((KrkNative*)AS_OBJECT(method))->flags & KRK_NATIVE_FLAGS_IS_DYNAMIC_PROPERTY) {
921  out = AS_NATIVE(method)->function(1, (KrkValue[]){krk_peek(0)}, 0);
922  } else if (((KrkNative*)AS_OBJECT(method))->flags & KRK_NATIVE_FLAGS_IS_CLASS_METHOD) {
923  out = OBJECT_VAL(krk_newBoundMethod(OBJECT_VAL(_class), AS_OBJECT(method)));
924  } else if (((KrkNative*)AS_OBJECT(method))->flags & KRK_NATIVE_FLAGS_IS_STATIC_METHOD) {
925  out = method;
926  } else {
927  out = OBJECT_VAL(krk_newBoundMethod(krk_peek(0), AS_OBJECT(method)));
928  }
929  } else if (IS_CLOSURE(method)) {
930  if (AS_CLOSURE(method)->flags & KRK_FUNCTION_FLAGS_IS_CLASS_METHOD) {
931  out = OBJECT_VAL(krk_newBoundMethod(OBJECT_VAL(_class), AS_OBJECT(method)));
932  } else if (AS_CLOSURE(method)->flags & KRK_FUNCTION_FLAGS_IS_STATIC_METHOD) {
933  out = method;
934  } else {
935  out = OBJECT_VAL(krk_newBoundMethod(krk_peek(0), AS_OBJECT(method)));
936  }
937  } else {
938  /* Does it have a descriptor __get__? */
939  KrkClass * type = krk_getType(method);
940  if (type->_descget) {
941  krk_push(method);
942  krk_swap(1);
943  krk_push(krk_callSimple(OBJECT_VAL(type->_descget), 2, 0));
944  return 1;
945  }
946  out = method;
947  }
948  krk_pop();
949  krk_push(out);
950  return 1;
951 }

1507 static int handleException() {
1508  int stackOffset, frameOffset;
1509  int exitSlot = (krk_currentThread.exitOnFrame >= 0) ? krk_currentThread.frames[krk_currentThread.exitOnFrame].outSlots : 0;
1510  for (stackOffset = (int)(krk_currentThread.stackTop - krk_currentThread.stack - 1);
1511  stackOffset >= exitSlot &&
1512  !IS_TRY_HANDLER(krk_currentThread.stack[stackOffset]) &&
1513  !IS_WITH_HANDLER(krk_currentThread.stack[stackOffset]) &&
1514  !IS_EXCEPT_HANDLER(krk_currentThread.stack[stackOffset])
1515  ; stackOffset--);
1516  if (stackOffset < exitSlot) {
1517  if (exitSlot == 0) {
1518  /*
1519  * No exception was found and we have reached the top of the call stack.
1520  * Call dumpTraceback to present the exception to the user and reset the
1521  * VM stack state. It should still be safe to execute more code after
1522  * this reset, so the repl can throw errors and keep accepting new lines.
1523  */
1524  if (!(vm.globalFlags & KRK_GLOBAL_CLEAN_OUTPUT)) krk_dumpTraceback();
1525  krk_currentThread.frameCount = 0;
1526  }
1527  /* If exitSlot was not 0, there was an exception during a call to runNext();
1528  * this is likely to be raised higher up the stack as an exception in the outer
1529  * call, but we don't want to print the traceback here. */
1530  return 1;
1531  }
1532 
1533  /* Find the call frame that owns this stack slot */
1534  for (frameOffset = krk_currentThread.frameCount - 1; frameOffset >= 0 && (int)krk_currentThread.frames[frameOffset].slots > stackOffset; frameOffset--);
1535  if (frameOffset == -1) {
1536  abort();
1537  }
1538 
1539  /* We found an exception handler and can reset the VM to its call frame. */
1540  closeUpvalues(stackOffset);
1541  krk_currentThread.stackTop = krk_currentThread.stack + stackOffset + 1;
1542  krk_currentThread.frameCount = frameOffset + 1;
1543 
1544  /* Clear the exception flag so we can continue executing from the handler. */
1545  krk_currentThread.flags &= ~KRK_THREAD_HAS_EXCEPTION;
1546  return 0;
1547 }
