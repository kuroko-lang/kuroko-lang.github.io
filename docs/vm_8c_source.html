169 void krk_dumpTraceback() {
170  if (!krk_valuesEqual(krk_currentThread.currentException,NONE_VAL())) {
171  krk_push(krk_currentThread.currentException);
172  KrkValue tracebackEntries;
173  if (IS_INSTANCE(krk_currentThread.currentException)
174  && krk_tableGet(&AS_INSTANCE(krk_currentThread.currentException)->fields, OBJECT_VAL(S("traceback")), &tracebackEntries)
175  && IS_list(tracebackEntries) && AS_LIST(tracebackEntries)->count > 0) {
176  /* This exception has a traceback we can print. */
177  fprintf(stderr, "Traceback (most recent call last):\n");
178  for (size_t i = 0; i < AS_LIST(tracebackEntries)->count; ++i) {
179 
180  /* Quietly skip invalid entries as we don't want to bother printing explanatory text for them */
181  if (!IS_TUPLE(AS_LIST(tracebackEntries)->values[i])) continue;
182  KrkTuple * entry = AS_TUPLE(AS_LIST(tracebackEntries)->values[i]);
183  if (entry->values.count != 2) continue;
184  if (!IS_CLOSURE(entry->values.values[0])) continue;
185  if (!IS_INTEGER(entry->values.values[1])) continue;
186 
187  /* Get the function and instruction index from this traceback entry */
188  KrkClosure * closure = AS_CLOSURE(entry->values.values[0]);
189  KrkFunction * function = closure->function;
190  size_t instruction = AS_INTEGER(entry->values.values[1]);
191 
192  /* Calculate the line number */
193  int lineNo = (int)krk_lineNumber(&function->chunk, instruction);
194 
195  /* Print the simple stuff that we already know */
196  fprintf(stderr, " File \"%s\", line %d, in %s\n",
197  (function->chunk.filename ? function->chunk.filename->chars : "?"),
198  lineNo,
199  (function->name ? function->name->chars : "(unnamed)"));
200 
201  /* Try to open the file */
202  if (function->chunk.filename) {
203  FILE * f = fopen(function->chunk.filename->chars, "r");
204  if (f) {
205  int line = 1;
206  do {
207  char c = fgetc(f);
208  if (c < -1) break;
209  if (c == '\n') {
210  line++;
211  continue;
212  }
213  if (line == lineNo) {
214  fprintf(stderr," ");
215  while (c == ' ') c = fgetc(f);
216  do {
217  fputc(c, stderr);
218  c = fgetc(f);
219  } while (!feof(f) && c > 0 && c != '\n');
220  fprintf(stderr, "\n");
221  break;
222  }
223  } while (!feof(f));
224  fclose(f);
225  }
226  }
227  }
228  }
229 
230  /* Is this a SyntaxError? Handle those specially. */
231  if (krk_isInstanceOf(krk_currentThread.currentException, vm.exceptions->syntaxError)) {
232  KrkValue result = krk_callSimple(OBJECT_VAL(krk_getType(krk_currentThread.currentException)->_tostr), 1, 0);
233  fprintf(stderr, "%s\n", AS_CSTRING(result));
234  return;
235  }
236  /* Clear the exception state while printing the exception. */
237  krk_currentThread.flags &= ~(KRK_THREAD_HAS_EXCEPTION);
238  fprintf(stderr, "%s", krk_typeName(krk_currentThread.currentException));
239  KrkValue result = krk_callSimple(OBJECT_VAL(krk_getType(krk_currentThread.currentException)->_tostr), 1, 0);
240  if (!IS_STRING(result)) {
241  fprintf(stderr, "\n");
242  } else {
243  fprintf(stderr, ": %s\n", AS_CSTRING(result));
244  }
245  /* Turn the exception flag back on */
246  krk_currentThread.flags |= KRK_THREAD_HAS_EXCEPTION;
247  }

253 static void attachTraceback(void) {
254  if (IS_INSTANCE(krk_currentThread.currentException)) {
255  KrkInstance * theException = AS_INSTANCE(krk_currentThread.currentException);
256 
257  /* If there already is a traceback, don't add a new one; this exception was re-raised. */
258  KrkValue existing;
259  if (krk_tableGet(&theException->fields, OBJECT_VAL(S("traceback")), &existing)) return;
260 
261  KrkValue tracebackList = krk_list_of(0,NULL,0);
262  krk_push(tracebackList);
263  /* Build the traceback object */
264  if (krk_currentThread.frameCount) {
265  for (size_t i = 0; i < krk_currentThread.frameCount; i++) {
266  KrkCallFrame * frame = &krk_currentThread.frames[i];
267  KrkTuple * tbEntry = krk_newTuple(2);
268  krk_push(OBJECT_VAL(tbEntry));
269  tbEntry->values.values[tbEntry->values.count++] = OBJECT_VAL(frame->closure);
270  tbEntry->values.values[tbEntry->values.count++] = INTEGER_VAL(frame->ip - frame->closure->function->chunk.code - 1);
271  krk_tupleUpdateHash(tbEntry);
272  krk_writeValueArray(AS_LIST(tracebackList), OBJECT_VAL(tbEntry));
273  krk_pop();
274  }
275  }
276  krk_attachNamedValue(&theException->fields, "traceback", tracebackList);
277  krk_pop();
278  } /* else: probably a legacy 'raise str', just don't bother. */
279 }

287  char buf[1024] = {0};

289  va_start(args, fmt);
290  size_t len = vsnprintf(buf, 1024, fmt, args);

307  attachTraceback();

589 int krk_processComplexArguments(int argCount, KrkValueArray * positionals, KrkTable * keywords) {
590  size_t kwargsCount = AS_INTEGER(krk_currentThread.stackTop[-1]);
591  krk_pop(); /* Pop the arg counter */
592  argCount--;
593 
594  krk_initValueArray(positionals);
595  krk_initTable(keywords);
596 
597  /* First, process all the positionals, including any from extractions. */
598  size_t existingPositionalArgs = argCount - kwargsCount * 2;
599  for (size_t i = 0; i < existingPositionalArgs; ++i) {
600  krk_writeValueArray(positionals, krk_currentThread.stackTop[-argCount + i]);
601  }
602 
603  KrkValue * startOfExtras = &krk_currentThread.stackTop[-kwargsCount * 2];
604  /* Now unpack everything else. */
605  for (size_t i = 0; i < kwargsCount; ++i) {
606  KrkValue key = startOfExtras[i*2];
607  KrkValue value = startOfExtras[i*2 + 1];
608  if (IS_KWARGS(key)) {
609  if (AS_INTEGER(key) == LONG_MAX-1) { /* unpack list */
610  unpackIterableFast(value);
611  } else if (AS_INTEGER(key) == LONG_MAX-2) { /* unpack dict */
612  if (!IS_INSTANCE(value)) {
613  krk_runtimeError(vm.exceptions->typeError, "**expression value is not a dict.");
614  return 0;
615  }
616  for (size_t i = 0; i < AS_DICT(value)->capacity; ++i) {
617  KrkTableEntry * entry = &AS_DICT(value)->entries[i];
618  if (entry->key.type != VAL_KWARGS) {
619  if (!IS_STRING(entry->key)) {
620  krk_runtimeError(vm.exceptions->typeError, "**expression contains non-string key");
621  return 0;
622  }
623  if (!krk_tableSet(keywords, entry->key, entry->value)) {
624  krk_runtimeError(vm.exceptions->typeError, "got multiple values for argument '%s'", AS_CSTRING(entry->key));
625  return 0;
626  }
627  }
628  }
629  } else if (AS_INTEGER(key) == LONG_MAX) { /* single value */
630  krk_writeValueArray(positionals, value);

653 static int call(KrkClosure * closure, int argCount, int extra) {
654  KrkValue * startOfPositionals = &krk_currentThread.stackTop[-argCount];
655  size_t potentialPositionalArgs = closure->function->requiredArgs + closure->function->keywordArgs;
656  size_t totalArguments = closure->function->requiredArgs + closure->function->keywordArgs + closure->function->collectsArguments + closure->function->collectsKeywords;
657  size_t offsetOfExtraArgs = closure->function->requiredArgs + closure->function->keywordArgs;
658  size_t argCountX = argCount;
659  KrkValueArray * positionals;
660  KrkTable * keywords;
661 
662  if (argCount && IS_KWARGS(krk_currentThread.stackTop[-1])) {
663 
664  KrkValue myList = krk_list_of(0,NULL,0);
665  krk_currentThread.scratchSpace[0] = myList;
666  KrkValue myDict = krk_dict_of(0,NULL,0);
667  krk_currentThread.scratchSpace[1] = myDict;
668  positionals = AS_LIST(myList);
669  keywords = AS_DICT(myDict);
670 
671  /* This processes the existing argument list into a ValueArray and a Table with the args and keywords */
672  if (!krk_processComplexArguments(argCount, positionals, keywords)) goto _errorDuringPositionals;
673  argCount--; /* It popped the KWARGS value from the top, so we have one less argument */
674 
675  /* Do we already know we have too many arguments? Let's bail before doing a bunch of work. */
676  if ((positionals->count > potentialPositionalArgs) && (!closure->function->collectsArguments)) {
677  checkArgumentCount(closure,positionals->count);
678  goto _errorDuringPositionals;
679  }
680 
681  /* Prepare stack space for all potential positionals, mark them unset */
682  for (size_t i = 0; i < (size_t)argCount; ++i) {
683  krk_currentThread.stackTop[-argCount + i] = KWARGS_VAL(0);
684  }
685 
686  /* Do we have a bunch of unused keyword argument slots? Fill them in. */
687  while ((size_t)argCount < potentialPositionalArgs) {
688  krk_push(KWARGS_VAL(0));
689  argCount++;
690  }
691 
692  /* Did we have way more arguments than we needed? Put the stack where it should be. */
693  while ((size_t)argCount > potentialPositionalArgs) {
694  krk_pop();
695  argCount--;
696  }
697 
698  /* Place positional arguments */
699  for (size_t i = 0; i < potentialPositionalArgs && i < positionals->count; ++i) {
700  krk_currentThread.stackTop[-argCount + i] = positionals->values[i];
701  }
702 
703  if (closure->function->collectsArguments) {
704  size_t count = (positionals->count > potentialPositionalArgs) ? (positionals->count - potentialPositionalArgs) : 0;
705  KrkValue * offset = (count == 0) ? NULL : &positionals->values[potentialPositionalArgs];
706  krk_push(krk_list_of(count, offset, 0));
707  argCount++;
708  }
709 
710  krk_freeValueArray(positionals);
711  krk_currentThread.scratchSpace[0] = NONE_VAL();
712 
713  /* Now place keyword arguments */
714  for (size_t i = 0; i < keywords->capacity; ++i) {
715  KrkTableEntry * entry = &keywords->entries[i];
716  if (entry->key.type != VAL_KWARGS) {
717  KrkValue name = entry->key;
718  KrkValue value = entry->value;
719  /* See if we can place it */
720  for (int j = 0; j < (int)closure->function->requiredArgs; ++j) {
721  if (krk_valuesEqual(name, closure->function->requiredArgNames.values[j])) {
722  if (!IS_KWARGS(krk_currentThread.stackTop[-argCount + j])) {
723  multipleDefs(closure,j);
724  goto _errorAfterPositionals;
725  }
726  krk_currentThread.stackTop[-argCount + j] = value;
727  goto _finishKwarg;
728  }
729  }
730  /* See if it's a keyword arg. */
731  for (int j = 0; j < (int)closure->function->keywordArgs; ++j) {
732  if (krk_valuesEqual(name, closure->function->keywordArgNames.values[j])) {
733  if (!IS_KWARGS(krk_currentThread.stackTop[-argCount + j + closure->function->requiredArgs])) {
734  multipleDefs(closure, j + closure->function->requiredArgs);
735  goto _errorAfterPositionals;
736  }
737  krk_currentThread.stackTop[-argCount + j + closure->function->requiredArgs] = value;
738  goto _finishKwarg;
739  }
740  }
741  if (!closure->function->collectsKeywords) {
742  krk_runtimeError(vm.exceptions->typeError, "%s() got an unexpected keyword argument '%s'",
743  closure->function->name ? closure->function->name->chars : "<unnamed function>",
744  AS_CSTRING(name));
745  goto _errorAfterPositionals;
746  }
747  continue;
748 _finishKwarg:
749  entry->key = KWARGS_VAL(0);
750  entry->value = BOOLEAN_VAL(1);
751  continue;
752  }
753  }
754 
755  /* If this function takes a **kwargs, we need to provide it as a dict */
756  if (closure->function->collectsKeywords) {
757  krk_push(krk_dict_of(0,NULL,0));
758  argCount++;
759  krk_tableAddAll(keywords, AS_DICT(krk_peek(0)));
760  }
761 
762  krk_freeTable(keywords);
763  krk_currentThread.scratchSpace[1] = NONE_VAL();
764 
765  for (size_t i = 0; i < (size_t)closure->function->requiredArgs; ++i) {
766  if (IS_KWARGS(krk_currentThread.stackTop[-argCount + i])) {
767  krk_runtimeError(vm.exceptions->typeError, "%s() missing required positional argument: '%s'",
768  closure->function->name ? closure->function->name->chars : "<unnamed function>",
769  AS_CSTRING(closure->function->requiredArgNames.values[i]));
770  goto _errorAfterKeywords;
771  }
772  }
773 
774  argCountX = argCount - (closure->function->collectsArguments + closure->function->collectsKeywords);
775  } else {
776  /* We can't have had any kwargs. */
777  if ((size_t)argCount > potentialPositionalArgs && closure->function->collectsArguments) {
778  krk_push(NONE_VAL()); krk_push(NONE_VAL()); krk_pop(); krk_pop();
779  startOfPositionals[offsetOfExtraArgs] = krk_list_of(argCount - potentialPositionalArgs,
780  &startOfPositionals[potentialPositionalArgs], 0);
781  argCount = closure->function->requiredArgs + 1;
782  argCountX = argCount - 1;
783  while (krk_currentThread.stackTop > startOfPositionals + argCount) krk_pop();
784  }
785  }
786  if (!checkArgumentCount(closure, argCountX)) {
787  return 0;
788  }
789  while (argCount < (int)totalArguments) {
790  krk_push(KWARGS_VAL(0));
791  argCount++;
792  }
793  if (krk_currentThread.frameCount == KRK_CALL_FRAMES_MAX) {
794  krk_runtimeError(vm.exceptions->baseException, "Too many call frames.");
795  return 0;
796  }
797  KrkCallFrame * frame = &krk_currentThread.frames[krk_currentThread.frameCount++];
798  frame->closure = closure;
799  frame->ip = closure->function->chunk.code;
800  frame->slots = (krk_currentThread.stackTop - argCount) - krk_currentThread.stack;
801  frame->outSlots = (krk_currentThread.stackTop - argCount - extra) - krk_currentThread.stack;
802  frame->globals = &closure->function->globalsContext->fields;
803  return 1;
804 
805 _errorDuringPositionals:
806  krk_freeValueArray(positionals);
807  krk_currentThread.scratchSpace[0] = NONE_VAL();
808 _errorAfterPositionals:
809  krk_freeTable(keywords);
810  krk_currentThread.scratchSpace[1] = NONE_VAL();
811 _errorAfterKeywords:
812  return 0;
813 }

834 int krk_callValue(KrkValue callee, int argCount, int extra) {
835  if (IS_OBJECT(callee)) {
836  switch (OBJECT_TYPE(callee)) {
837  case OBJ_CLOSURE:
838  return call(AS_CLOSURE(callee), argCount, extra);
839  case OBJ_NATIVE: {
840  NativeFn native = (NativeFn)AS_NATIVE(callee)->function;
841  if (argCount && IS_KWARGS(krk_currentThread.stackTop[-1])) {
842  KrkValue myList = krk_list_of(0,NULL,0);
843  krk_currentThread.scratchSpace[0] = myList;
844  KrkValue myDict = krk_dict_of(0,NULL,0);
845  krk_currentThread.scratchSpace[1] = myDict;
846  if (!krk_processComplexArguments(argCount, AS_LIST(myList), AS_DICT(myDict))) {
847  return 0;
848  }
849  argCount--; /* Because that popped the kwargs value */
850  krk_currentThread.stackTop -= argCount + extra; /* We can just put the stack back to normal */
851  krk_push(myList);
852  krk_push(myDict);
853  krk_currentThread.scratchSpace[0] = NONE_VAL();
854  krk_currentThread.scratchSpace[1] = NONE_VAL();
855  krk_writeValueArray(AS_LIST(myList), myDict);
856  KrkValue result = native(AS_LIST(myList)->count-1, AS_LIST(myList)->values, 1);
857  if (krk_currentThread.stackTop == krk_currentThread.stack) return 0;
858  krk_pop();
859  krk_pop();
860  krk_push(result);
861  } else {
862  KrkValue * stackCopy = malloc(argCount * sizeof(KrkValue));
863  memcpy(stackCopy, krk_currentThread.stackTop - argCount, argCount * sizeof(KrkValue));
864  KrkValue result = native(argCount, stackCopy, 0);
865  free(stackCopy);
866  if (krk_currentThread.stackTop == krk_currentThread.stack) return 0;
867  krk_currentThread.stackTop -= argCount + extra;
868  krk_push(result);
869  }
870  return 2;
871  }
872  case OBJ_INSTANCE: {
873  KrkClass * _class = AS_INSTANCE(callee)->_class;
874  KrkValue callFunction;
875  if (_class->_call) {
876  return krk_callValue(OBJECT_VAL(_class->_call), argCount + 1, 0);
877  } else if (krk_tableGet(&_class->methods, vm.specialMethodNames[METHOD_CALL], &callFunction)) {
878  return krk_callValue(callFunction, argCount + 1, 0);
879  } else {
880  krk_runtimeError(vm.exceptions->typeError, "Attempted to call non-callable type: %s", krk_typeName(callee));
881  return 0;
882  }
883  }
884  case OBJ_CLASS: {
885  KrkClass * _class = AS_CLASS(callee);
886  KrkInstance * newInstance = krk_newInstance(_class);
887  krk_currentThread.stackTop[-argCount - 1] = OBJECT_VAL(newInstance);
888  KrkValue initializer;
889  if (_class->_init) {
890  return krk_callValue(OBJECT_VAL(_class->_init), argCount + 1, 0);
891  } else if (krk_tableGet(&_class->methods, vm.specialMethodNames[METHOD_INIT], &initializer)) {
892  return krk_callValue(initializer, argCount + 1, 0);
893  } else if (argCount != 0) {
894  krk_runtimeError(vm.exceptions->attributeError, "Class does not have an __init__ but arguments were passed to initializer: %d", argCount);
895  return 0;
896  }
897  return 1;
898  }
899  case OBJ_BOUND_METHOD: {
900  KrkBoundMethod * bound = AS_BOUND_METHOD(callee);
901  krk_currentThread.stackTop[-argCount - 1] = bound->receiver;
902  if (!bound->method) {
903  krk_runtimeError(vm.exceptions->argumentError, "Attempted to call a method binding with no attached callable (did you forget to return something from a method decorator?)");
904  return 0;
905  }
906  return krk_callValue(OBJECT_VAL(bound->method), argCount + 1, 0);
907  }
908  default:
909  break;
910  }
911  }
912  krk_runtimeError(vm.exceptions->typeError, "Attempted to call non-callable type: %s", krk_typeName(callee));
913  return 0;
914 }

1395 static int handleException() {
1396  int stackOffset, frameOffset;
1397  int exitSlot = (krk_currentThread.exitOnFrame >= 0) ? krk_currentThread.frames[krk_currentThread.exitOnFrame].outSlots : 0;
1398  for (stackOffset = (int)(krk_currentThread.stackTop - krk_currentThread.stack - 1); stackOffset >= exitSlot && !IS_TRY_HANDLER(krk_currentThread.stack[stackOffset]); stackOffset--);
1399  if (stackOffset < exitSlot) {
1400  if (exitSlot == 0) {
1401  /*
1402  * No exception was found and we have reached the top of the call stack.
1403  * Call dumpTraceback to present the exception to the user and reset the
1404  * VM stack state. It should still be safe to execute more code after
1405  * this reset, so the repl can throw errors and keep accepting new lines.
1406  */
1407  krk_dumpTraceback();
1408  krk_resetStack();
1409  krk_currentThread.frameCount = 0;
1410  }
1411  /* If exitSlot was not 0, there was an exception during a call to runNext();
1412  * this is likely to be raised higher up the stack as an exception in the outer
1413  * call, but we don't want to print the traceback here. */
1414  return 1;
1415  }
1416 
1417  /* Find the call frame that owns this stack slot */
1418  for (frameOffset = krk_currentThread.frameCount - 1; frameOffset >= 0 && (int)krk_currentThread.frames[frameOffset].slots > stackOffset; frameOffset--);
1419  if (frameOffset == -1) {
1420  fprintf(stderr, "Internal error: Call stack is corrupted - unable to find\n");
1421  fprintf(stderr, " call frame that owns exception handler.\n");
1422  exit(1);
1423  }
1424 
1425  /* We found an exception handler and can reset the VM to its call frame. */
1426  closeUpvalues(stackOffset);
1427  krk_currentThread.stackTop = krk_currentThread.stack + stackOffset + 1;
1428  krk_currentThread.frameCount = frameOffset + 1;
1429 
1430  /* Clear the exception flag so we can continue executing from the handler. */
1431  krk_currentThread.flags &= ~KRK_THREAD_HAS_EXCEPTION;
1432  return 0;
1433 }
