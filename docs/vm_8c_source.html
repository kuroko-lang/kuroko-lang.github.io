278 static void attachTraceback(void) {
279  if (IS_INSTANCE(krk_currentThread.currentException)) {
280  KrkInstance * theException = AS_INSTANCE(krk_currentThread.currentException);
281  KrkValue tracebackList;
282  if (krk_tableGet(&theException->fields, OBJECT_VAL(S("traceback")), &tracebackList)) {
283  krk_push(tracebackList);
284  } else {
285  krk_push(NONE_VAL());
286  }
287  tracebackList = krk_list_of(0,NULL,0);
288  krk_push(tracebackList);
289 
290  /* Build the traceback object */
291  if (krk_currentThread.frameCount) {
292 
293  /* Go up until we get to the exit frame */
294  size_t frameOffset = 0;
295  if (krk_currentThread.stackTop > krk_currentThread.stack) {
296  size_t stackOffset = krk_currentThread.stackTop - krk_currentThread.stack - 1;
297  while (stackOffset > 0 && !IS_TRY_HANDLER(krk_currentThread.stack[stackOffset])) stackOffset--;
298  frameOffset = krk_currentThread.frameCount - 1;
299  while (frameOffset > 0 && krk_currentThread.frames[frameOffset].slots > stackOffset) frameOffset--;
300  }
301 
302  for (size_t i = frameOffset; i < krk_currentThread.frameCount; i++) {
303  KrkCallFrame * frame = &krk_currentThread.frames[i];
304  KrkTuple * tbEntry = krk_newTuple(2);
305  krk_push(OBJECT_VAL(tbEntry));
306  tbEntry->values.values[tbEntry->values.count++] = OBJECT_VAL(frame->closure);
307  tbEntry->values.values[tbEntry->values.count++] = INTEGER_VAL(frame->ip - frame->closure->function->chunk.code - 1);
308  krk_writeValueArray(AS_LIST(tracebackList), OBJECT_VAL(tbEntry));
309  krk_pop();
310  }
311  }
312 
313  if (IS_list(krk_peek(1))) {
314  KrkValueArray * existingTraceback = AS_LIST(krk_peek(1));
315  for (size_t i = 0; i < existingTraceback->count; ++i) {
316  krk_writeValueArray(AS_LIST(tracebackList), existingTraceback->values[i]);
317  }
318  }
319 
320  krk_attachNamedValue(&theException->fields, "traceback", tracebackList);
321  krk_pop();
322  krk_pop();
323  } /* else: probably a legacy 'raise str', just don't bother. */
324 }

632  for (size_t i = 0; i < existingPositionalArgs; ++i) {
633  krk_writeValueArray(positionals, krk_currentThread.stackTop[-argCount + i - TOP_ARGS]);
634  }
635 
636  KrkValue * startOfExtras = &krk_currentThread.stackTop[-kwargsCount * 2 - TOP_ARGS];
637  /* Now unpack everything else. */
638  for (size_t i = 0; i < kwargsCount; ++i) {
639  KrkValue key = startOfExtras[i*2];
640  KrkValue value = startOfExtras[i*2 + 1];
641  if (IS_KWARGS(key)) {
642  if (AS_INTEGER(key) == KWARGS_LIST) { /* unpack list */
643  unpackIterableFast(value);
644  } else if (AS_INTEGER(key) == KWARGS_DICT) { /* unpack dict */
645  if (!IS_dict(value)) {
646  krk_runtimeError(vm.exceptions->typeError, "%s(): **expression value is not a dict.", name);
647  return 0;
648  }
649  for (size_t i = 0; i < AS_DICT(value)->capacity; ++i) {
650  KrkTableEntry * entry = &AS_DICT(value)->entries[i];
651  if (!IS_KWARGS(entry->key)) {
652  if (!IS_STRING(entry->key)) {
653  krk_runtimeError(vm.exceptions->typeError, "%s(): **expression contains non-string key", name);
654  return 0;
655  }
656  if (!krk_tableSet(keywords, entry->key, entry->value)) {
657  krk_runtimeError(vm.exceptions->typeError, "%s() got multiple values for argument '%s'", name, AS_CSTRING(entry->key));
658  return 0;
659  }
660  }
661  }
662  } else if (AS_INTEGER(key) == KWARGS_SINGLE) { /* single value */
663  krk_writeValueArray(positionals, value);
664  }
665  } else if (IS_STRING(key)) {
666  if (!krk_tableSet(keywords, key, value)) {
667  krk_runtimeError(vm.exceptions->typeError, "%s() got multiple values for argument '%s'", name, AS_CSTRING(key));
668  return 0;
669  }
670  }
671  }
672  return 1;

686 static inline int _callManaged(KrkClosure * closure, int argCount, int returnDepth) {
687  size_t potentialPositionalArgs = closure->function->potentialPositionals;
688  size_t totalArguments = closure->function->totalArguments;
689  size_t offsetOfExtraArgs = potentialPositionalArgs;
690  size_t argCountX = argCount;
691 
692  if (argCount && unlikely(IS_KWARGS(krk_currentThread.stackTop[-1]))) {
693 
694  KrkValue myList = krk_list_of(0,NULL,0);
695  krk_push(myList);
696  KrkValueArray * positionals;
697  positionals = AS_LIST(myList);
698 
699  KrkValue myDict = krk_dict_of(0,NULL,0);
700  krk_push(myDict);
701  KrkTable * keywords;
702  keywords = AS_DICT(myDict);
703 
704  /* This processes the existing argument list into a ValueArray and a Table with the args and keywords */
705  if (!krk_processComplexArguments(argCount, positionals, keywords, closure->function->name ? closure->function->name->chars : "<unnamed>")) return 0;
706 
707  /* Store scratch while we adjust; we can not make calls while using these scratch
708  * registers as they may be clobbered by a nested call to _callManaged. */
709  krk_currentThread.scratchSpace[0] = myList;
710  krk_currentThread.scratchSpace[1] = myList;
711 
712  /* Pop three things, including the kwargs count */
713  krk_pop(); /* dict */
714  krk_pop(); /* list */
715  krk_pop(); /* kwargs */
716 
717  /* We popped the kwargs sentinel, which counted for one argCount */
718  argCount--;
719 
720  /* Do we already know we have too many arguments? Let's bail before doing a bunch of work. */
721  if ((positionals->count > potentialPositionalArgs) && !(closure->function->obj.flags & KRK_OBJ_FLAGS_CODEOBJECT_COLLECTS_ARGS)) {
722  checkArgumentCount(closure,positionals->count);
723  goto _errorDuringPositionals;
724  }
725 
726  /* Prepare stack space for all potential positionals, mark them unset */
727  for (size_t i = 0; i < (size_t)argCount; ++i) {
728  krk_currentThread.stackTop[-argCount + i] = KWARGS_VAL(0);
729  }
730 
731  /* Do we have a bunch of unused keyword argument slots? Fill them in. */
732  while ((size_t)argCount < potentialPositionalArgs) {
733  krk_push(KWARGS_VAL(0));
734  argCount++;
735  }
736 
737  /* Did we have way more arguments than we needed? Put the stack where it should be. */
738  while ((size_t)argCount > potentialPositionalArgs) {
739  krk_pop();
740  argCount--;
741  }
742 
743  /* Place positional arguments */
744  for (size_t i = 0; i < potentialPositionalArgs && i < positionals->count; ++i) {
745  krk_currentThread.stackTop[-argCount + i] = positionals->values[i];
746  }
747 
748  if (closure->function->obj.flags & KRK_OBJ_FLAGS_CODEOBJECT_COLLECTS_ARGS) {
749  size_t count = (positionals->count > potentialPositionalArgs) ? (positionals->count - potentialPositionalArgs) : 0;
750  KrkValue * offset = (count == 0) ? NULL : &positionals->values[potentialPositionalArgs];
751  krk_push(krk_list_of(count, offset, 0));
752  argCount++;
753  }
754 
755  /* We're done with the positionals */
756  krk_currentThread.scratchSpace[0] = NONE_VAL();
757 
758  /* Now place keyword arguments */
759  for (size_t i = 0; i < keywords->capacity; ++i) {
760  KrkTableEntry * entry = &keywords->entries[i];
761  if (!IS_KWARGS(entry->key)) {
762  KrkValue name = entry->key;
763  KrkValue value = entry->value;
764  /* See if we can place it */
765  for (int j = 0; j < (int)closure->function->requiredArgs; ++j) {
766  if (krk_valuesEqual(name, closure->function->requiredArgNames.values[j])) {
767  if (!IS_KWARGS(krk_currentThread.stackTop[-argCount + j])) {
768  multipleDefs(closure,j);
769  goto _errorAfterPositionals;
770  }
771  krk_currentThread.stackTop[-argCount + j] = value;
772  goto _finishKwarg;
773  }
774  }
775  /* See if it's a keyword arg. */
776  for (int j = 0; j < (int)closure->function->keywordArgs; ++j) {
777  if (krk_valuesEqual(name, closure->function->keywordArgNames.values[j])) {
778  if (!IS_KWARGS(krk_currentThread.stackTop[-argCount + j + closure->function->requiredArgs])) {
779  multipleDefs(closure, j + closure->function->requiredArgs);
780  goto _errorAfterPositionals;
781  }
782  krk_currentThread.stackTop[-argCount + j + closure->function->requiredArgs] = value;
783  goto _finishKwarg;
784  }
785  }
786  if (!(closure->function->obj.flags & KRK_OBJ_FLAGS_CODEOBJECT_COLLECTS_KWS)) {
787  krk_runtimeError(vm.exceptions->typeError, "%s() got an unexpected keyword argument '%s'",
788  closure->function->name ? closure->function->name->chars : "<unnamed>",
789  AS_CSTRING(name));
790  goto _errorAfterPositionals;
791  }
792  continue;
793 _finishKwarg:
794  entry->key = KWARGS_VAL(0);
795  entry->value = BOOLEAN_VAL(1);
796  continue;
797  }
798  }
799 
800  /* If this function takes a **kwargs, we need to provide it as a dict */
801  if (closure->function->obj.flags & KRK_OBJ_FLAGS_CODEOBJECT_COLLECTS_KWS) {
802  krk_push(krk_dict_of(0,NULL,0));
803  argCount++;
804  krk_tableAddAll(keywords, AS_DICT(krk_peek(0)));
805  }
806 
807  /* We're done with the keywords */
808  krk_currentThread.scratchSpace[1] = NONE_VAL();
809 
810  for (size_t i = 0; i < (size_t)closure->function->requiredArgs; ++i) {
811  if (IS_KWARGS(krk_currentThread.stackTop[-argCount + i])) {
812  krk_runtimeError(vm.exceptions->typeError, "%s() missing required positional argument: '%s'",
813  closure->function->name ? closure->function->name->chars : "<unnamed>",
814  AS_CSTRING(closure->function->requiredArgNames.values[i]));
815  goto _errorAfterKeywords;
816  }
817  }
818 
819  argCountX = argCount - (!!(closure->function->obj.flags & KRK_OBJ_FLAGS_CODEOBJECT_COLLECTS_ARGS) + !!(closure->function->obj.flags & KRK_OBJ_FLAGS_CODEOBJECT_COLLECTS_KWS));
820  } else if ((size_t)argCount > potentialPositionalArgs && (closure->function->obj.flags & KRK_OBJ_FLAGS_CODEOBJECT_COLLECTS_ARGS)) {
821  krk_push(NONE_VAL()); krk_push(NONE_VAL()); krk_pop(); krk_pop();
822  KrkValue * startOfPositionals = &krk_currentThread.stackTop[-argCount];
823  KrkValue tmp = krk_list_of(argCount - potentialPositionalArgs,
824  &startOfPositionals[potentialPositionalArgs], 0);
825  startOfPositionals = &krk_currentThread.stackTop[-argCount];
826  startOfPositionals[offsetOfExtraArgs] = tmp;
827  argCount = closure->function->requiredArgs + 1;
828  argCountX = argCount - 1;
829  while (krk_currentThread.stackTop > startOfPositionals + argCount) krk_pop();
830  }
831 
832  if (unlikely(!checkArgumentCount(closure, argCountX))) goto _errorAfterKeywords;
833 
834  while (argCount < (int)totalArguments) {
835  krk_push(KWARGS_VAL(0));
836  argCount++;
837  }
838 
839  if (unlikely(closure->function->obj.flags & (KRK_OBJ_FLAGS_CODEOBJECT_IS_GENERATOR | KRK_OBJ_FLAGS_CODEOBJECT_IS_COROUTINE))) {
840  KrkInstance * gen = krk_buildGenerator(closure, krk_currentThread.stackTop - argCount, argCount);
841  krk_currentThread.stackTop = krk_currentThread.stackTop - argCount - returnDepth;
842  krk_push(OBJECT_VAL(gen));
843  return 2;
844  }
845 
846  if (unlikely(krk_currentThread.frameCount == vm.maximumCallDepth)) {
847  krk_runtimeError(vm.exceptions->baseException, "maximum recursion depth exceeded");
848  goto _errorAfterKeywords;
849  }
850 
851  KrkCallFrame * frame = &krk_currentThread.frames[krk_currentThread.frameCount++];
852  frame->closure = closure;
853  frame->ip = closure->function->chunk.code;
854  frame->slots = (krk_currentThread.stackTop - argCount) - krk_currentThread.stack;
855  frame->outSlots = frame->slots - returnDepth;
856  frame->globals = &closure->function->globalsContext->fields;
857  FRAME_IN(frame);
858  return 1;
859 
860 _errorDuringPositionals:
861  krk_currentThread.scratchSpace[0] = NONE_VAL();
862 _errorAfterPositionals:
863  krk_currentThread.scratchSpace[1] = NONE_VAL();
864 _errorAfterKeywords:
865  return 0;
866 }

916  krk_push(myDict);
917

919  * our list + dict to be at the top: [kwargs] [list] [dict] */

923  krk_writeValueArray(AS_LIST(myList), myDict);
924

958 int krk_callValue(KrkValue callee, int argCount, int returnDepth) {
959  if (likely(IS_OBJECT(callee))) {
960  _innerObject:
961  switch (OBJECT_TYPE(callee)) {
962  case KRK_OBJ_CLOSURE: return _callManaged(AS_CLOSURE(callee), argCount, returnDepth);
963  case KRK_OBJ_NATIVE: return _callNative(AS_NATIVE(callee), argCount, returnDepth);
964  case KRK_OBJ_INSTANCE: {
965  KrkClass * _class = AS_INSTANCE(callee)->_class;
966  if (likely(_class->_call != NULL)) {
967  if (unlikely(returnDepth == 0)) _rotate(argCount);
968  krk_currentThread.stackTop[-argCount - 1] = callee;
969  argCount++;
970  returnDepth = returnDepth ? (returnDepth - 1) : 0;
971  return (_class->_call->type == KRK_OBJ_CLOSURE) ? _callManaged((KrkClosure*)_class->_call, argCount, returnDepth) : _callNative((KrkNative*)_class->_call, argCount, returnDepth);
972  } else {
973  krk_runtimeError(vm.exceptions->typeError, "'%s' object is not callable", krk_typeName(callee));
974  return 0;
975  }
976  }
977  case KRK_OBJ_CLASS: {
978  KrkClass * _class = AS_CLASS(callee);
979  KrkInstance * newInstance = krk_newInstance(_class);
980  if (likely(_class->_init != NULL)) {
981  if (unlikely(returnDepth == 0)) _rotate(argCount);
982  krk_currentThread.stackTop[-argCount - 1] = OBJECT_VAL(newInstance);
983  callee = OBJECT_VAL(_class->_init);
984  argCount++;
985  returnDepth = returnDepth ? (returnDepth - 1) : 0;
986  goto _innerObject;
987  } else if (unlikely(argCount != 0)) {
988  krk_runtimeError(vm.exceptions->typeError, "%s() takes no arguments (%d given)",
989  _class->name->chars, argCount);
990  return 0;
991  }
992  krk_currentThread.stackTop -= argCount + returnDepth;
993  krk_push(OBJECT_VAL(newInstance));
994  return 2;
995  }
996  case KRK_OBJ_BOUND_METHOD: {
997  KrkBoundMethod * bound = AS_BOUND_METHOD(callee);
998  if (unlikely(!bound->method)) {
999  krk_runtimeError(vm.exceptions->argumentError, "???");
1000  return 0;
1001  }
1002  if (unlikely(returnDepth == 0)) _rotate(argCount);
1003  krk_currentThread.stackTop[-argCount - 1] = bound->receiver;
1004  callee = OBJECT_VAL(bound->method);
1005  argCount++;
1006  returnDepth = returnDepth ? (returnDepth - 1) : 0;
1007  goto _innerObject;
1008  }
1009  default:
1010  break;
1011  }
1012  }
1013  krk_runtimeError(vm.exceptions->typeError, "'%s' object is not callable", krk_typeName(callee));
1014  return 0;
1015 }

1634 static int handleException() {
1635  int stackOffset, frameOffset;
1636  int exitSlot = (krk_currentThread.exitOnFrame >= 0) ? krk_currentThread.frames[krk_currentThread.exitOnFrame].outSlots : 0;
1637  for (stackOffset = (int)(krk_currentThread.stackTop - krk_currentThread.stack - 1);
1638  stackOffset >= exitSlot &&
1639  !IS_TRY_HANDLER(krk_currentThread.stack[stackOffset]) &&
1640  !IS_WITH_HANDLER(krk_currentThread.stack[stackOffset]) &&
1641  !IS_EXCEPT_HANDLER(krk_currentThread.stack[stackOffset])
1642  ; stackOffset--);
1643  if (stackOffset < exitSlot) {
1644  if (exitSlot == 0) {
1645  /*
1646  * No exception was found and we have reached the top of the call stack.
1647  * Call dumpTraceback to present the exception to the user and reset the
1648  * VM stack state. It should still be safe to execute more code after
1649  * this reset, so the repl can throw errors and keep accepting new lines.
1650  */
1651  if (!(vm.globalFlags & KRK_GLOBAL_CLEAN_OUTPUT)) krk_dumpTraceback();
1652  }
1653  krk_currentThread.frameCount = krk_currentThread.exitOnFrame;
1654 
1655  /* Ensure stack is in the expected place, as if we returned None. */
1656  krk_currentThread.stackTop = &krk_currentThread.stack[exitSlot];
1657 
1658  /* If exitSlot was not 0, there was an exception during a call to runNext();
1659  * this is likely to be raised higher up the stack as an exception in the outer
1660  * call, but we don't want to print the traceback here. */
1661  return 1;
1662  }
1663 
1664  /* Find the call frame that owns this stack slot */
1665  for (frameOffset = krk_currentThread.frameCount - 1; frameOffset >= 0 && (int)krk_currentThread.frames[frameOffset].slots > stackOffset; frameOffset--);
1666  if (frameOffset == -1) {
1667  abort();
1668  }
1669 
1670  /* We found an exception handler and can reset the VM to its call frame. */
1671  closeUpvalues(stackOffset);
1672  krk_currentThread.stackTop = krk_currentThread.stack + stackOffset + 1;
1673  krk_currentThread.frameCount = frameOffset + 1;
1674 
1675  /* Clear the exception flag so we can continue executing from the handler. */
1676  krk_currentThread.flags &= ~KRK_THREAD_HAS_EXCEPTION;
1677  return 0;
1678 }
