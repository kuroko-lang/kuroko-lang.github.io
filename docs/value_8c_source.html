52 void krk_printValueSafe(FILE * f, KrkValue printable) {
53  if (!IS_OBJECT(printable)) {
54  switch (KRK_VAL_TYPE(printable)) {
55  case KRK_VAL_INTEGER: fprintf(f, PRIkrk_int, AS_INTEGER(printable)); break;
56  case KRK_VAL_BOOLEAN: fprintf(f, "%s", AS_BOOLEAN(printable) ? "True" : "False"); break;
57  case KRK_VAL_NONE: fprintf(f, "None"); break;
58  case KRK_VAL_HANDLER:
59  switch (AS_HANDLER_TYPE(printable)) {
60  case OP_PUSH_TRY: fprintf(f, "{try->%d}", (int)AS_HANDLER_TARGET(printable)); break;
61  case OP_PUSH_WITH: fprintf(f, "{with->%d}", (int)AS_HANDLER_TARGET(printable)); break;
62  case OP_RAISE: fprintf(f, "{raise<-%d}", (int)AS_HANDLER_TARGET(printable)); break;
63  case OP_FILTER_EXCEPT: fprintf(f, "{except<-%d}", (int)AS_HANDLER_TARGET(printable)); break;
64  case OP_BEGIN_FINALLY: fprintf(f, "{finally<-%d}", (int)AS_HANDLER_TARGET(printable)); break;
65  case OP_RETURN: fprintf(f, "{return<-%d}", (int)AS_HANDLER_TARGET(printable)); break;
66  case OP_END_FINALLY: fprintf(f, "{end<-%d}", (int)AS_HANDLER_TARGET(printable)); break;
67  case OP_EXIT_LOOP: fprintf(f, "{exit<-%d}", (int)AS_HANDLER_TARGET(printable)); break;
68  }
69  break;
70  case KRK_VAL_KWARGS: {
71  if (AS_INTEGER(printable) == KWARGS_SINGLE) {
72  fprintf(f, "{unpack single}");
73  } else if (AS_INTEGER(printable) == KWARGS_LIST) {
74  fprintf(f, "{unpack list}");
75  } else if (AS_INTEGER(printable) == KWARGS_DICT) {
76  fprintf(f, "{unpack dict}");
77  } else if (AS_INTEGER(printable) == KWARGS_NIL) {
78  fprintf(f, "{unpack nil}");
79  } else if (AS_INTEGER(printable) == KWARGS_UNSET) {
80  fprintf(f, "{unset default}");
81  } else {
82  fprintf(f, "{sentinel=" PRIkrk_int "}",AS_INTEGER(printable));
83  }
84  break;
85  }
86  default:
87  if (IS_FLOATING(printable)) fprintf(f, "%.16g", AS_FLOATING(printable));
88  break;
89  }
90  } else if (IS_STRING(printable)) {
91  fprintf(f, "'");
92  /*
93  * Print at most STRING_DEBUG_TRUNCATE characters, as bytes, escaping anything not ASCII.
94  * See also str.__repr__ which does something similar with escape sequences, but this
95  * is a dumber, safer, and slightly faster approach.
96  */
97  for (size_t c = 0; c < AS_STRING(printable)->length && c < STRING_DEBUG_TRUNCATE; ++c) {
98  unsigned char byte = (unsigned char)AS_CSTRING(printable)[c];
99  switch (byte) {
100  case '\\': fprintf(f, "\\\\"); break;
101  case '\n': fprintf(f, "\\n"); break;
102  case '\r': fprintf(f, "\\r"); break;
103  case '\'': fprintf(f, "\\'"); break;
104  default: {
105  if (byte < ' ' || byte > '~') {
106  fprintf(f, "\\x%02x", byte);
107  } else {
108  fprintf(f, "%c", byte);
109  }
110  break;
111  }
112  }
113  }
114  if (AS_STRING(printable)->length > STRING_DEBUG_TRUNCATE) {
115  fprintf(f,"...");
116  }
117  fprintf(f,"'");
118  } else {
119  switch (AS_OBJECT(printable)->type) {
120  case KRK_OBJ_CODEOBJECT: fprintf(f, "<function %s>", AS_codeobject(printable)->name ? AS_codeobject(printable)->name->chars : "?"); break;
121  case KRK_OBJ_CLASS: fprintf(f, "<class %s>", AS_CLASS(printable)->name ? AS_CLASS(printable)->name->chars : "?"); break;
122  case KRK_OBJ_INSTANCE: fprintf(f, "<instance of %s>", AS_INSTANCE(printable)->_class->name->chars); break;
123  case KRK_OBJ_NATIVE: fprintf(f, "<nativefn %s>", ((KrkNative*)AS_OBJECT(printable))->name); break;
124  case KRK_OBJ_CLOSURE: fprintf(f, "<function %s>", AS_CLOSURE(printable)->function->name->chars); break;
125  case KRK_OBJ_BYTES: fprintf(f, "<bytes of len %ld>", (long)AS_BYTES(printable)->length); break;
126  case KRK_OBJ_TUPLE: {
127  fprintf(f, "(");
128  for (size_t i = 0; i < AS_TUPLE(printable)->values.count; ++i) {
129  krk_printValueSafe(f, AS_TUPLE(printable)->values.values[i]);
130  if (i + 1 != AS_TUPLE(printable)->values.count) {
131  fprintf(f, ",");
132  }
133  }
134  fprintf(f, ")");
135  } break;
136  case KRK_OBJ_BOUND_METHOD: fprintf(f, "<method %s>",
137  AS_BOUND_METHOD(printable)->method ? (
138  AS_BOUND_METHOD(printable)->method->type == KRK_OBJ_CLOSURE ? ((KrkClosure*)AS_BOUND_METHOD(printable)->method)->function->name->chars :
139  (AS_BOUND_METHOD(printable)->method->type == KRK_OBJ_NATIVE ? ((KrkNative*)AS_BOUND_METHOD(printable)->method)->name : "(unknown)")) : "(corrupt bound method)"); break;
140  default: fprintf(f, "<%s>", krk_typeName(printable)); break;
141  }
142  }
143 }

152 static inline int _krk_method_equivalence(KrkValue a, KrkValue b) {
153  KrkClass * type = krk_getType(a);
154  if (likely(type && type->_eq)) {
155  krk_push(a);
156  krk_push(b);
157  KrkValue result = krk_callDirect(type->_eq,2);
158  if (unlikely(krk_currentThread.flags & KRK_THREAD_HAS_EXCEPTION)) return 0;
159  if (IS_BOOLEAN(result)) return AS_BOOLEAN(result);
160  if (!IS_NOTIMPL(result)) return !krk_isFalsey(result);
161  }
162 
163  type = krk_getType(b);
164  if (type && type->_eq) {
165  krk_push(b);
166  krk_push(a);
167  KrkValue result = krk_callDirect(type->_eq,2);
168  if (unlikely(krk_currentThread.flags & KRK_THREAD_HAS_EXCEPTION)) return 0;
169  if (IS_BOOLEAN(result)) return AS_BOOLEAN(result);
170  if (!IS_NOTIMPL(result)) return !krk_isFalsey(result);
171  }
172 
173  return 0;
174 }
