64 static void * _startthread(void * _threadObj) {
65  memset(&krk_currentThread, 0, sizeof(KrkThreadState));
66  krk_currentThread.frames = calloc(KRK_CALL_FRAMES_MAX,sizeof(KrkCallFrame));
67 
68  _obtain_lock(_threadLock);
69  if (vm.threads->next) {
70  krk_currentThread.next = vm.threads->next;
71  }
72  vm.threads->next = &krk_currentThread;
73  _release_lock(_threadLock);
74 
75  /* Get our run function */
76  struct Thread * self = _threadObj;
77  self->threadState = &krk_currentThread;
78  self->tid = gettid();
79 
80  KrkValue runMethod = NONE_VAL();
81  KrkClass * ourType = self->inst._class;
82  if (!krk_tableGet(&ourType->methods, OBJECT_VAL(S("run")), &runMethod)) {
83  krk_runtimeError(ThreadError, "Thread object has no run() method");
84  } else {
85  krk_push(OBJECT_VAL(self));
86  krk_callValue(runMethod, 1, 0);
87  krk_runNext();
88  }
89 
90  self->alive = 0;
91 
92  /* Remove this thread from the thread pool, its stack is garbage anyway */
93  _obtain_lock(_threadLock);
94  krk_resetStack();
95  KrkThreadState * previous = vm.threads;
96  while (previous) {
97  if (previous->next == &krk_currentThread) {
98  previous->next = krk_currentThread.next;
99  break;
100  }
101  previous = previous->next;
102  }
103  _release_lock(_threadLock);
104 
105  return NULL;
106 }
