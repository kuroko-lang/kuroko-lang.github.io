68 static void * _startthread(void * _threadObj) {
69  memset(&krk_currentThread, 0, sizeof(KrkThreadState));
70  krk_currentThread.frames = calloc(vm.maximumCallDepth,sizeof(KrkCallFrame));
71  vm.globalFlags |= KRK_GLOBAL_THREADS;
72  _obtain_lock(_threadLock);
73  if (vm.threads->next) {
74  krk_currentThread.next = vm.threads->next;
75  }
76  vm.threads->next = &krk_currentThread;
77  _release_lock(_threadLock);
78 
79  /* Get our run function */
80  struct Thread * self = _threadObj;
81  self->threadState = &krk_currentThread;
82  self->tid = gettid();
83 
84  KrkValue runMethod = NONE_VAL();
85  KrkClass * ourType = self->inst._class;
86  if (!krk_tableGet(&ourType->methods, OBJECT_VAL(S("run")), &runMethod)) {
87  krk_runtimeError(ThreadError, "Thread object has no run() method");
88  } else {
89  krk_push(runMethod);
90  krk_push(OBJECT_VAL(self));
91  krk_callStack(1);
92  }
93 
94  self->alive = 0;
95 
96  /* Remove this thread from the thread pool, its stack is garbage anyway */
97  _obtain_lock(_threadLock);
98  krk_resetStack();
99  KrkThreadState * previous = vm.threads;
100  while (previous) {
101  if (previous->next == &krk_currentThread) {
102  previous->next = krk_currentThread.next;
103  break;
104  }
105  previous = previous->next;
106  }
107  _release_lock(_threadLock);
108 
109  FREE_ARRAY(size_t, krk_currentThread.stack, krk_currentThread.stackSize);
110  free(krk_currentThread.frames);
111 
112  return NULL;
113 }
