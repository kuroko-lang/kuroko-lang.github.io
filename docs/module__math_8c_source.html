21  if (!IS_FLOATING(arg)) { switch (KRK_VAL_TYPE(arg)) { \
22  case KRK_VAL_INTEGER: arg = FLOATING_VAL(AS_INTEGER(arg)); break; \

122 static KrkValue _math_frexp(int argc, KrkValue argv[], int hasKw) {
123  ONE_ARGUMENT(frexp)
124  FORCE_FLOAT(argv[0])
125  if (!IS_FLOATING(argv[0])) {
126  REAL_NUMBER_NOT(frexp,argv[0])
127  }
128  int exp = 0;
129  double result = frexp(AS_FLOATING(argv[0]), &exp);
130  KrkTuple * outValue = krk_newTuple(2);
131  outValue->values.values[0] = FLOATING_VAL(result);
132  outValue->values.values[1] = INTEGER_VAL(exp);
133  outValue->values.count = 2;
134  return OBJECT_VAL(outValue);
135 }
