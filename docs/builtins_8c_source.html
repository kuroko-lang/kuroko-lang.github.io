16  if (argc != 1) return krk_runtimeError(vm.exceptions->argumentError, "wrong number of arguments or bad type, got %d\n", argc);
17 
18  /* Create a new list instance */
19  KrkValue myList = krk_list_of(0,NULL,0);
20  krk_push(myList);
21 
22  if (IS_INSTANCE(argv[0])) {
23  /* Obtain self-reference */
24  KrkInstance * self = AS_INSTANCE(argv[0]);
25  for (size_t i = 0; i < self->fields.capacity; ++i) {
26  if (self->fields.entries[i].key.type != VAL_KWARGS) {
27 
28  krk_writeValueArray(AS_LIST(myList),
29  self->fields.entries[i].key);
30  }
31  }
32  } else if (IS_CLASS(argv[0])) {
33  KrkClass * _class = AS_CLASS(argv[0]);
34  while (_class) {
35  for (size_t i = 0; i < _class->methods.capacity; ++i) {
36  if (_class->methods.entries[i].key.type != VAL_KWARGS) {
37  krk_writeValueArray(AS_LIST(myList),
38  _class->methods.entries[i].key);
39  }
40  }
41  _class = _class->base;
42  }
43  }
44 
45  KrkClass * type = krk_getType(argv[0]);
46 
47  while (type) {
48  for (size_t i = 0; i < type->methods.capacity; ++i) {
49  if (type->methods.entries[i].key.type != VAL_KWARGS) {
50  krk_writeValueArray(AS_LIST(myList),
51  type->methods.entries[i].key);
52  }
53  }
54  type = type->base;
55  }
56 
57  /* Throw it at a set to get unique, unordered results */
58  krk_push(krk_set_of(AS_LIST(myList)->count, AS_LIST(myList)->values, 0));
59  krk_swap(1);
60  krk_pop();

64  krk_push(myList);
65  krk_swap(1);

556  KrkCallFrame * frame = &krk_currentThread.frames[krk_currentThread.frameCount-index];
557  KrkFunction * func = frame->closure->function;
558  size_t offset = frame->ip - func->chunk.code;
559 
560  /* First, we'll populate with arguments */
561  size_t slot = 0;
562  for (short int i = 0; i < func->requiredArgs; ++i) {
563  krk_tableSet(AS_DICT(dict),
564  func->requiredArgNames.values[i],
565  krk_currentThread.stack[frame->slots + slot]);
566  slot++;
567  }
568  for (short int i = 0; i < func->keywordArgs; ++i) {
569  krk_tableSet(AS_DICT(dict),
570  func->keywordArgNames.values[i],
571  krk_currentThread.stack[frame->slots + slot]);
572  slot++;
573  }
574  if (func->collectsArguments) {
575  krk_tableSet(AS_DICT(dict),
576  func->requiredArgNames.values[func->requiredArgs],
577  krk_currentThread.stack[frame->slots + slot]);
578  slot++;
579  }
580  if (func->collectsKeywords) {
581  krk_tableSet(AS_DICT(dict),
582  func->keywordArgNames.values[func->keywordArgs],
583  krk_currentThread.stack[frame->slots + slot]);
584  slot++;
585  }
586  /* Now we need to find out what non-argument locals are valid... */
587  for (size_t i = 0; i < func->localNameCount; ++i) {
588  if (func->localNames[i].birthday <= offset &&
589  func->localNames[i].deathday >= offset) {
590  krk_tableSet(AS_DICT(dict),
591  OBJECT_VAL(func->localNames[i].name),
592  krk_currentThread.stack[frame->slots + func->localNames[i].id]);
593  }
594  }
