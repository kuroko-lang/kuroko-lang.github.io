10 KrkValue krk_dirObject(int argc, KrkValue argv[], int hasKw) {
11  if (argc != 1) return krk_runtimeError(vm.exceptions->argumentError, "wrong number of arguments or bad type, got %d\n", argc);
12 
13  /* Create a new list instance */
14  KrkValue myList = krk_list_of(0,NULL,0);
15  krk_push(myList);
16 
17  if (IS_INSTANCE(argv[0])) {
18  /* Obtain self-reference */
19  KrkInstance * self = AS_INSTANCE(argv[0]);
20 
21  /* First add each method of the class */
22  for (size_t i = 0; i < self->_class->methods.capacity; ++i) {
23  if (self->_class->methods.entries[i].key.type != VAL_KWARGS) {
24  krk_writeValueArray(AS_LIST(myList),
25  self->_class->methods.entries[i].key);
26  }
27  }
28 
29  /* Then add each field of the instance */
30  for (size_t i = 0; i < self->fields.capacity; ++i) {
31  if (self->fields.entries[i].key.type != VAL_KWARGS) {
32  krk_writeValueArray(AS_LIST(myList),
33  self->fields.entries[i].key);
34  }
35  }
36  } else {
37  if (IS_CLASS(argv[0])) {
38  KrkClass * _class = AS_CLASS(argv[0]);
39  for (size_t i = 0; i < _class->methods.capacity; ++i) {
40  if (_class->methods.entries[i].key.type != VAL_KWARGS) {
41  krk_writeValueArray(AS_LIST(myList),
42  _class->methods.entries[i].key);
43  }
44  }
45  for (size_t i = 0; i < _class->fields.capacity; ++i) {
46  if (_class->fields.entries[i].key.type != VAL_KWARGS) {
47  krk_writeValueArray(AS_LIST(myList),
48  _class->fields.entries[i].key);
49  }
50  }
51  }
52  KrkClass * type = krk_getType(argv[0]);
53 
54  for (size_t i = 0; i < type->methods.capacity; ++i) {
55  if (type->methods.entries[i].key.type != VAL_KWARGS) {
56  krk_writeValueArray(AS_LIST(myList),
57  type->methods.entries[i].key);
58  }
59  }
60  }
61 
62  /* Prepare output value */
63  krk_pop();
64  return myList;
65 }

154 static KrkValue _print(int argc, KrkValue argv[], int hasKw) {
155  KrkValue sepVal, endVal;
156  char * sep = " "; size_t sepLen = 1;
157  char * end = "\n"; size_t endLen = 1;
158  if (hasKw) {
159  if (krk_tableGet(AS_DICT(argv[argc]), OBJECT_VAL(S("sep")), &sepVal)) {
160  if (!IS_STRING(sepVal)) return krk_runtimeError(vm.exceptions->typeError, "'sep' should be a string, not '%s'", krk_typeName(sepVal));
161  sep = AS_CSTRING(sepVal);
162  sepLen = AS_STRING(sepVal)->length;
163  }
164  if (krk_tableGet(AS_DICT(argv[argc]), OBJECT_VAL(S("end")), &endVal)) {
165  if (!IS_STRING(endVal)) return krk_runtimeError(vm.exceptions->typeError, "'end' should be a string, not '%s'", krk_typeName(endVal));
166  end = AS_CSTRING(endVal);
167  endLen = AS_STRING(endVal)->length;
168  }
169  }
170  for (int i = 0; i < argc; ++i) {
171  KrkValue printable = argv[i];
172  if (IS_STRING(printable)) { /* krk_printValue runs repr */
173  /* Make sure we handle nil bits correctly. */
174  for (size_t j = 0; j < AS_STRING(printable)->length; ++j) {
175  fputc(AS_CSTRING(printable)[j], stdout);
176  }
177  } else {
178  krk_printValue(stdout, printable);
179  }
180  char * thingToPrint = (i == argc - 1) ? end : sep;
181  for (size_t j = 0; j < ((i == argc - 1) ? endLen : sepLen); ++j) {
182  fputc(thingToPrint[j], stdout);
183  }
184  }
185  return NONE_VAL();
186 }
