25 inline int krk_hashValue(KrkValue value, uint32_t *hashOut) {
26  switch (KRK_VAL_TYPE(value)) {
27  case KRK_VAL_BOOLEAN:
28  case KRK_VAL_INTEGER:
29  case KRK_VAL_NONE:
30  case KRK_VAL_HANDLER:
31  case KRK_VAL_KWARGS:
32  *hashOut = (uint32_t)AS_INTEGER(value);
33  return 0;
34  case KRK_VAL_OBJECT:
35  if (AS_OBJECT(value)->flags & KRK_OBJ_FLAGS_VALID_HASH) {
36  *hashOut = AS_OBJECT(value)->hash;
37  return 0;
38  }
39  break;
40  default:
41  *hashOut = (uint32_t)AS_FLOATING(value);
42  return 0;
43  }
44  KrkClass * type = krk_getType(value);
45  if (type && type->_hash) {
46  krk_push(value);
47  KrkValue result = krk_callDirect(type->_hash, 1);
48  if (!IS_INTEGER(result)) goto _unhashable;
49  *hashOut = (uint32_t)AS_INTEGER(result);
50  return 0;
51  }
52  if (IS_CLASS(value)) {
53  *hashOut = INTEGER_VAL((int)(intptr_t)AS_OBJECT(value));
54  return 0;
55  }
56 _unhashable:
57  if (IS_NONE(krk_currentThread.currentException))
58  krk_runtimeError(vm.exceptions->typeError, "unhashable type: '%T'", value);
59  return 1;
60 }

62 KrkTableEntry * krk_findEntry(KrkTableEntry * entries, size_t capacity, KrkValue key) {
63  uint32_t index;
64  if (krk_hashValue(key, &index)) {
65  return NULL;
66  }
67  index &= (capacity-1);
68  KrkTableEntry * tombstone = NULL;
69  for (;;) {
70  KrkTableEntry * entry = &entries[index];
71  if (entry->key == KWARGS_VAL(0)) {
72  return tombstone != NULL ? tombstone : entry;
73  } else if (entry->key == KWARGS_VAL(1)) {
74  if (tombstone == entry) return tombstone;
75  if (tombstone == NULL) tombstone = entry;
76  } else if (krk_valuesSameOrEqual(entry->key, key)) {
77  return entry;
78  }
79  index = (index + 1) & (capacity-1);
80  }
81 }

112 void krk_tableAdjustCapacity(KrkTable * table, size_t capacity) {
113  if (capacity) {
114  /* Fast power-of-two calculation */
115  size_t powerOfTwoCapacity = __builtin_clz(1) - __builtin_clz(capacity);
116  if ((1UL << powerOfTwoCapacity) != capacity) powerOfTwoCapacity++;
117  capacity = (1UL << powerOfTwoCapacity);
118  }
119 
120  KrkTableEntry * entries = ALLOCATE(KrkTableEntry, capacity);
121  for (size_t i = 0; i < capacity; ++i) {
122  entries[i].key = KWARGS_VAL(0);
123  entries[i].value = KWARGS_VAL(0);
124  }
125 
126  table->count = 0;
127  for (size_t i = 0; i < table->capacity; ++i) {
128  KrkTableEntry * entry = &table->entries[i];
129  if (IS_KWARGS(entry->key)) continue;
130  KrkTableEntry * dest = krk_findEntry(entries, capacity, entry->key);
131  dest->key = entry->key;
132  dest->value = entry->value;
133  table->count++;
134  }
135 
136  FREE_ARRAY(KrkTableEntry, table->entries, table->capacity);
137  table->entries = entries;
138  table->capacity = capacity;
139 }
