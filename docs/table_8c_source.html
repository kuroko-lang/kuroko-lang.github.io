25 inline int krk_hashValue(KrkValue value, uint32_t *hashOut) {
26  switch (KRK_VAL_TYPE(value)) {
27  case KRK_VAL_BOOLEAN:
28  case KRK_VAL_INTEGER:
29  case KRK_VAL_NONE:
30  case KRK_VAL_HANDLER:
31  case KRK_VAL_KWARGS:
32  *hashOut = (uint32_t)AS_INTEGER(value);
33  return 0;
34  case KRK_VAL_OBJECT:
35  if (AS_OBJECT(value)->flags & KRK_OBJ_FLAGS_VALID_HASH) {
36  *hashOut = AS_OBJECT(value)->hash;
37  return 0;
38  }
39  break;
40  default:
41  *hashOut = (uint32_t)AS_FLOATING(value);
42  return 0;
43  }
44  KrkClass * type = krk_getType(value);
45  if (type && type->_hash) {
46  krk_push(value);
47  KrkValue result = krk_callSimple(OBJECT_VAL(type->_hash), 1, 0);
48  if (!IS_INTEGER(result)) goto _unhashable;
49  *hashOut = (uint32_t)AS_INTEGER(result);
50  return 0;
51  }
52 _unhashable:
53  krk_runtimeError(vm.exceptions->typeError, "unhashable type: '%s'", krk_typeName(value));
54  return 1;
55 }

57 KrkTableEntry * krk_findEntry(KrkTableEntry * entries, size_t capacity, KrkValue key) {
58  uint32_t index;
59  if (krk_hashValue(key, &index)) {
60  return NULL;
61  }
62  index %= capacity;
63  KrkTableEntry * tombstone = NULL;
64  for (;;) {
65  KrkTableEntry * entry = &entries[index];
66  if (IS_KWARGS(entry->key)) {
67  if (IS_NONE(entry->value)) {
68  return tombstone != NULL ? tombstone : entry;
69  } else {
70  if (tombstone == NULL) tombstone = entry;
71  }
72  } else if (krk_valuesEqual(entry->key, key)) {
73  return entry;
74  }
75  index = (index + 1) % capacity;
76  }
77 }
