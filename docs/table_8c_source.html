24 inline uint32_t krk_hashValue(KrkValue value) {
25  switch (value.type) {
26  case KRK_VAL_INTEGER: return (uint32_t)(AS_INTEGER(value));
27  case KRK_VAL_FLOATING: return (uint32_t)(AS_FLOATING(value) * 1000); /* arbitrary; what's a good way to hash floats? */
28  case KRK_VAL_BOOLEAN: return (uint32_t)(AS_BOOLEAN(value));
29  case KRK_VAL_NONE: return 0;
30  case KRK_VAL_OBJECT: return (uint32_t)(AS_OBJECT(value))->hash;
31  default: return 0;
32  }
33 }

35 KrkTableEntry * krk_findEntry(KrkTableEntry * entries, size_t capacity, KrkValue key) {
36  uint32_t index = krk_hashValue(key) % capacity;
37  KrkTableEntry * tombstone = NULL;
38  for (;;) {
39  KrkTableEntry * entry = &entries[index];
40  if (entry->key.type == KRK_VAL_KWARGS) {
41  if (IS_NONE(entry->value)) {
42  return tombstone != NULL ? tombstone : entry;
43  } else {
44  if (tombstone == NULL) tombstone = entry;
45  }
46  } else if (krk_valuesEqual(entry->key, key)) {
47  return entry;
48  }
49  index = (index + 1) % capacity;
50  }
51 }

53 static void adjustCapacity(KrkTable * table, size_t capacity) {
54  KrkTableEntry * entries = ALLOCATE(KrkTableEntry, capacity);
55  for (size_t i = 0; i < capacity; ++i) {
56  entries[i].key = KWARGS_VAL(0);
57  entries[i].value = NONE_VAL();
58  }
59 
60  table->count = 0;
61  for (size_t i = 0; i < table->capacity; ++i) {
62  KrkTableEntry * entry = &table->entries[i];
63  if (entry->key.type == KRK_VAL_KWARGS) continue;
64  KrkTableEntry * dest = krk_findEntry(entries, capacity, entry->key);
65  dest->key = entry->key;
66  dest->value = entry->value;
67  table->count++;
68  }
69 
70  FREE_ARRAY(KrkTableEntry, table->entries, table->capacity);
71  table->entries = entries;
72  table->capacity = capacity;
73 }

75 int krk_tableSet(KrkTable * table, KrkValue key, KrkValue value) {
76  if (table->count + 1 > table->capacity * TABLE_MAX_LOAD) {
77  size_t capacity = GROW_CAPACITY(table->capacity);
78  adjustCapacity(table, capacity);
79  }
80  KrkTableEntry * entry = krk_findEntry(table->entries, table->capacity, key);
81  int isNewKey = entry->key.type == KRK_VAL_KWARGS;
82  if (isNewKey && IS_NONE(entry->value)) table->count++;
83  entry->key = key;
84  entry->value = value;
85  return isNewKey;
86 }
