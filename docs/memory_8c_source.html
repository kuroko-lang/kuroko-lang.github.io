8 void * krk_reallocate(void * ptr, size_t old, size_t new) {
9  vm.bytesAllocated += new - old;
10 
11  if (new > old && ptr != krk_currentThread.stack && &krk_currentThread == vm.threads && !(vm.globalFlags & KRK_GLOBAL_GC_PAUSED)) {
12 #ifdef ENABLE_STRESS_GC
13  if (vm.globalFlags & KRK_GLOBAL_ENABLE_STRESS_GC) {
14  krk_collectGarbage();
15  }
16 #endif
17  if (vm.bytesAllocated > vm.nextGC) {
18  krk_collectGarbage();
19  }
20  }
21 
22  if (new == 0) {
23  free(ptr);
24  return NULL;
25  }
26 
27  return realloc(ptr, new);

30 static void freeObject(KrkObj * object) {
31  switch (object->type) {
32  case KRK_OBJ_STRING: {
33  KrkString * string = (KrkString*)object;
34  FREE_ARRAY(char, string->chars, string->length + 1);
35  if (string->codes && string->codes != string->chars) free(string->codes);
36  FREE(KrkString, object);
37  break;
38  }
39  case KRK_OBJ_CODEOBJECT: {
40  KrkCodeObject * function = (KrkCodeObject*)object;
41  krk_freeChunk(&function->chunk);
42  krk_freeValueArray(&function->requiredArgNames);
43  krk_freeValueArray(&function->keywordArgNames);
44  FREE_ARRAY(KrkLocalEntry, function->localNames, function->localNameCount);
45  function->localNameCount = 0;
46  FREE(KrkCodeObject, object);
47  break;
48  }
49  case KRK_OBJ_NATIVE: {
50  FREE(KrkNative, object);
51  break;
52  }
53  case KRK_OBJ_CLOSURE: {
54  KrkClosure * closure = (KrkClosure*)object;
55  FREE_ARRAY(KrkUpvalue*,closure->upvalues,closure->upvalueCount);
56  krk_freeTable(&closure->fields);
57  FREE(KrkClosure, object);
58  break;
59  }
60  case KRK_OBJ_UPVALUE: {
61  FREE(KrkUpvalue, object);
62  break;
63  }
64  case KRK_OBJ_CLASS: {
65  KrkClass * _class = (KrkClass*)object;
66  krk_freeTable(&_class->methods);
67  FREE(KrkClass, object);
68  break;
69  }
70  case KRK_OBJ_INSTANCE: {
71  if (((KrkInstance*)object)->_class->_ongcsweep) ((KrkInstance*)object)->_class->_ongcsweep((KrkInstance*)object);
72  krk_freeTable(&((KrkInstance*)object)->fields);
73  FREE(KrkInstance, object);
74  break;
75  }
76  case KRK_OBJ_BOUND_METHOD:
77  FREE(KrkBoundMethod, object);
78  break;
79  case KRK_OBJ_TUPLE: {
80  KrkTuple * tuple = (KrkTuple*)object;
81  krk_freeValueArray(&tuple->values);
82  FREE(KrkTuple, object);
83  break;
84  }
85  case KRK_OBJ_BYTES: {
86  KrkBytes * bytes = (KrkBytes*)object;
87  FREE_ARRAY(uint8_t, bytes->bytes, bytes->length);
88  FREE(KrkBytes, bytes);
89  break;
90  }
91  }

142 static void blackenObject(KrkObj * object) {
143  switch (object->type) {
144  case KRK_OBJ_CLOSURE: {
145  KrkClosure * closure = (KrkClosure *)object;
146  krk_markObject((KrkObj*)closure->function);
147  for (size_t i = 0; i < closure->upvalueCount; ++i) {
148  krk_markObject((KrkObj*)closure->upvalues[i]);
149  }
150  krk_markValue(closure->annotations);
151  krk_markTable(&closure->fields);
152  break;
153  }
154  case KRK_OBJ_CODEOBJECT: {
155  KrkCodeObject * function = (KrkCodeObject *)object;
156  krk_markObject((KrkObj*)function->name);
157  krk_markObject((KrkObj*)function->qualname);
158  krk_markObject((KrkObj*)function->docstring);
159  krk_markObject((KrkObj*)function->chunk.filename);
160  krk_markObject((KrkObj*)function->globalsContext);
161  markArray(&function->requiredArgNames);
162  markArray(&function->keywordArgNames);
163  markArray(&function->chunk.constants);
164  for (size_t i = 0; i < function->localNameCount; ++i) {
165  krk_markObject((KrkObj*)function->localNames[i].name);
166  }
167  break;
168  }
169  case KRK_OBJ_UPVALUE:
170  krk_markValue(((KrkUpvalue*)object)->closed);
171  break;
172  case KRK_OBJ_CLASS: {
173  KrkClass * _class = (KrkClass *)object;
174  krk_markObject((KrkObj*)_class->name);
175  krk_markObject((KrkObj*)_class->filename);
176  krk_markObject((KrkObj*)_class->docstring);
177  krk_markObject((KrkObj*)_class->base);
178  krk_markTable(&_class->methods);
179  break;
180  }
181  case KRK_OBJ_INSTANCE: {
182  krk_markObject((KrkObj*)((KrkInstance*)object)->_class);
183  if (((KrkInstance*)object)->_class->_ongcscan) ((KrkInstance*)object)->_class->_ongcscan((KrkInstance*)object);
184  krk_markTable(&((KrkInstance*)object)->fields);
185  break;
186  }
187  case KRK_OBJ_BOUND_METHOD: {
188  KrkBoundMethod * bound = (KrkBoundMethod *)object;
189  krk_markValue(bound->receiver);
190  krk_markObject((KrkObj*)bound->method);
191  break;
192  }
193  case KRK_OBJ_TUPLE: {
194  KrkTuple * tuple = (KrkTuple *)object;
195  markArray(&tuple->values);
196  break;
197  }
198  case KRK_OBJ_NATIVE:
199  case KRK_OBJ_STRING:
200  case KRK_OBJ_BYTES:
201  break;
202  }

212 static size_t sweep() {
213  KrkObj * previous = NULL;
214  KrkObj * object = vm.objects;
215  size_t count = 0;
216  while (object) {
217  if (object->isMarked || object->isImmortal) {
218  object->isMarked = 0;
219  object->generation = 0;
220  previous = object;
221  object = object->next;
222  } else if (object->generation == 3) {
223  KrkObj * unreached = object;
224  object = object->next;
225  if (previous != NULL) {
226  previous->next = object;
227  } else {
228  vm.objects = object;
229  }
230  freeObject(unreached);
231  count++;
232  } else {
233  object->generation++;
234  previous = object;
235  object = object->next;
236  }
237  }
238  return count;
239 }
