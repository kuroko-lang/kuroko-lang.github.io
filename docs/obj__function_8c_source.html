16 static KrkTuple * functionArgs(KrkCodeObject * _self) {
17  KrkTuple * tuple = krk_newTuple(_self->requiredArgs + _self->keywordArgs + !!(_self->obj.flags & KRK_OBJ_FLAGS_CODEOBJECT_COLLECTS_ARGS) + !!(_self->obj.flags & KRK_OBJ_FLAGS_CODEOBJECT_COLLECTS_KWS));
18  krk_push(OBJECT_VAL(tuple));
19 
20  for (short i = 0; i < _self->requiredArgs; ++i) {
21  tuple->values.values[tuple->values.count++] = _self->requiredArgNames.values[i];
22  }
23 
24  for (short i = 0; i < _self->keywordArgs; ++i) {
25  struct StringBuilder sb = {0};
26  pushStringBuilderStr(&sb, AS_CSTRING(_self->keywordArgNames.values[i]), AS_STRING(_self->keywordArgNames.values[i])->length);
27  pushStringBuilder(&sb,'=');
28  tuple->values.values[tuple->values.count++] = finishStringBuilder(&sb);
29  }
30 
31  if (_self->obj.flags & KRK_OBJ_FLAGS_CODEOBJECT_COLLECTS_ARGS) {
32  struct StringBuilder sb = {0};
33  pushStringBuilder(&sb, '*');
34  pushStringBuilderStr(&sb, AS_CSTRING(_self->requiredArgNames.values[_self->requiredArgs]), AS_STRING(_self->requiredArgNames.values[_self->requiredArgs])->length);
35  tuple->values.values[tuple->values.count++] = finishStringBuilder(&sb);
36  }
37 
38  if (_self->obj.flags & KRK_OBJ_FLAGS_CODEOBJECT_COLLECTS_KWS) {
39  struct StringBuilder sb = {0};
40  pushStringBuilder(&sb, '*');
41  pushStringBuilder(&sb, '*');
42  pushStringBuilderStr(&sb, AS_CSTRING(_self->keywordArgNames.values[_self->keywordArgs]), AS_STRING(_self->keywordArgNames.values[_self->keywordArgs])->length);
43  tuple->values.values[tuple->values.count++] = finishStringBuilder(&sb);
44  }
45 
46  krk_pop();
47  return tuple;
48 }

59 FUNC_SIG(function,__init__) {
60  static __attribute__ ((unused)) const char* _method_name = "__init__";
61  METHOD_TAKES_EXACTLY(3);
62  CHECK_ARG(1,codeobject,KrkCodeObject*,code);
63 
64  if (!IS_INSTANCE(argv[3]))
65  return TYPE_ERROR(dict or instance object,argv[3]);
66 
67  if (IS_CLOSURE(argv[2]) && AS_CLOSURE(argv[2])->upvalueCount == code->upvalueCount) {
68  /* Option 1: A function with the same upvalue count. Copy the upvalues exactly.
69  * As an example, this can be a lambda with a bunch of unused upvalue
70  * references - like "lambda: a, b, c". These variables will be captured
71  * using the relevant scope - and we don't have to care about whether
72  * they were properly marked, because the compiler took care of it
73  * when the lambda was compiled.
74  */
75  krk_push(OBJECT_VAL(krk_newClosure(code, argv[3])));
76  memcpy(AS_CLOSURE(krk_peek(0))->upvalues, AS_CLOSURE(argv[2])->upvalues,
77  sizeof(KrkUpvalue*) * code->upvalueCount);
78  return krk_pop();
79  } else if (IS_TUPLE(argv[2]) && AS_TUPLE(argv[2])->values.count == code->upvalueCount) {
80  /* Option 2: A tuple of values. New upvalue containers are built for each value,
81  * but they are immediately closed with the value in the tuple. They
82  * exist independently for this closure instance, and are not shared with
83  * any other closures.
84  */
85  krk_push(OBJECT_VAL(krk_newClosure(code, argv[3])));
86  for (size_t i = 0; i < code->upvalueCount; ++i) {
87  AS_CLOSURE(krk_peek(0))->upvalues[i] = krk_newUpvalue(-1);
88  AS_CLOSURE(krk_peek(0))->upvalues[i]->closed = AS_TUPLE(argv[2])->values.values[i];
89  }
90  return krk_pop();
91  }
92 
93  return TYPE_ERROR(managed function with equal upvalue count or tuple,argv[2]);
94 }
